//program to count total number of notes in entered amount using loop 

   import java.util.Scanner;

public class CountNotes {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // Input the total amount
        System.out.print("Enter the amount: ");
        int amount = sc.nextInt();

        int[] notes = {2000, 500, 200, 100, 50, 20, 10, 5, 2, 1}; // available denominations
        int count;

        System.out.println("\nCurrency Count:");
        for (int i = 0; i < notes.length; i++) {
            if (amount >= notes[i]) {
                count = amount / notes[i];   // number of notes of this denomination
                amount = amount % notes[i];  // remaining amount
                System.out.println(notes[i] + " : " + count);
            }
        }

        sc.close();
    }
}
