// print numbers from 1 to 10 using loop 

   public class PrintNumbers {
    public static void main(String[] args) {
        // Using for loop
        System.out.println("Numbers from 1 to 10:");

        for (int i = 1; i <= 10; i++) {
            System.out.println(i);
        }
    }
}
