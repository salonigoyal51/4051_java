//create an abstract class Animal with an abstract method called sound(). 

      // Abstract base class
abstract class Animal {
    // Abstract method (no body)
    abstract void sound();

    // Concrete (non-abstract) method
    void eat() {
        System.out.println("All animals eat food.");
    }
}

// Subclass: Lion
class Lion extends Animal {
    // Implementing abstract method
    @Override
    void sound() {
        System.out.println("Lion roars: Roarrr!");
    }
}

// Subclass: Tiger
class Tiger extends Animal {
    // Implementing abstract method
    @Override
    void sound() {
        System.out.println("Tiger growls: Grrr!");
    }
}

// Main class
public class AbstractClassExample {
    public static void main(String[] args) {
        // You cannot create an object of an abstract class
        // Animal a = new Animal(); ‚ùå

        // Create objects of subclasses
        Animal lion = new Lion();
        Animal tiger = new Tiger();

        // Call methods
        lion.eat();
        lion.sound();

        tiger.eat();
        tiger.sound();
    }
}
