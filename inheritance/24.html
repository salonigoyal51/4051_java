//create an abstract class Shape2D with abstract methods draw() and resize(). 

   // Abstract base class
abstract class Shape2D {
    String shapeName;

    // Constructor
    Shape2D(String shapeName) {
        this.shapeName = shapeName;
    }

    // Abstract methods
    abstract void draw();
    abstract void resize(double factor);

    // Concrete method
    void displayShape() {
        System.out.println("Shape: " + shapeName);
    }
}

// Subclass: Rectangle
class Rectangle extends Shape2D {
    double length, width;

    Rectangle(double length, double width) {
        super("Rectangle");
        this.length = length;
        this.width = width;
    }

    @Override
    void draw() {
        System.out.println("Drawing a Rectangle with length " + length + " and width " + width + ".");
    }

    @Override
    void resize(double factor) {
        length *= factor;
        width *= factor;
        System.out.println("Resized Rectangle to new dimensions — Length: " + length + ", Width: " + width);
    }
}

// Subclass: Circle
class Circle extends Shape2D {
    double radius;

    Circle(double radius) {
        super("Circle");
        this.radius = radius;
    }

    @Override
    void draw() {
        System.out.println("Drawing a Circle with radius " + radius + ".");
    }

    @Override
    void resize(double factor) {
        radius *= factor;
        System.out.println("Resized Circle to new radius: " + radius);
    }
}

// Main class
public class AbstractShape2DExample {
    public static void main(String[] args) {
        // Create Rectangle object
        Shape2D rectangle = new Rectangle(10, 5);
        System.out.println("---- Rectangle Details ----");
        rectangle.displayShape();
        rectangle.draw();
        rectangle.resize(1.5); // Resize by 50%

        System.out.println();

        // Create Circle object
        Shape2D circle = new Circle(7);
        System.out.println("---- Circle Details ----");
        circle.displayShape();
        circle.draw();
        circle.resize(2); // Double the size
    }
}
