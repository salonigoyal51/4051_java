PROGRAM

//create a class called Animal with a method named move(). 

   // Base class
class Animal {
    void move() {
        System.out.println("Animals can move");
    }
}

// Subclass
class Cheetah extends Animal {
    // Overriding the move() method
    @Override
    void move() {
        System.out.println("Cheetah runs fast");
    }
}

// Main class to test the program
public class AnimalExample {
    public static void main(String[] args) {
        // Create an Animal object
        Animal a = new Animal();
        a.move(); // Calls Animal's move()

        // Create a Cheetah object
        Cheetah c = new Cheetah();
        c.move(); // Calls Cheetah's overridden move()
    }
}

OUTPUT

Animals can move
Cheetah runs fast
