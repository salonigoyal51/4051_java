//Write a Java program to create an abstract class GeometricShape with abstract methods area() and perimeter(). 

    import java.util.Scanner;

// Abstract class
abstract class GeometricShape {
    // Abstract methods (no body)
    abstract double area();
    abstract double perimeter();
}

// Subclass: Triangle
class Triangle extends GeometricShape {
    double side1, side2, side3;

    // Constructor
    Triangle(double s1, double s2, double s3) {
        side1 = s1;
        side2 = s2;
        side3 = s3;
    }

    // Calculate area using Heron's formula
    double area() {
        double s = (side1 + side2 + side3) / 2;
        return Math.sqrt(s * (s - side1) * (s - side2) * (s - side3));
    }

    // Calculate perimeter
    double perimeter() {
        return side1 + side2 + side3;
    }
}

// Subclass: Square
class Square extends GeometricShape {
    double side;

    // Constructor
    Square(double s) {
        side = s;
    }

    // Calculate area
    double area() {
        return side * side;
    }

    // Calculate perimeter
    double perimeter() {
        return 4 * side;
    }
}

// Main class
public class GeometricShapeTest {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // Triangle input
        System.out.println("Enter sides of the triangle:");
        double s1 = sc.nextDouble();
        double s2 = sc.nextDouble();
        double s3 = sc.nextDouble();
        Triangle t = new Triangle(s1, s2, s3);

        // Square input
        System.out.println("\nEnter side of the square:");
        double side = sc.nextDouble();
        Square sq = new Square(side);

        // Display results
        System.out.println("\n--- Results ---");
        System.out.println("Triangle Area: " + t.area());
        System.out.println("Triangle Perimeter: " + t.perimeter());
        System.out.println("Square Area: " + sq.area());
        System.out.println("Square Perimeter: " + sq.perimeter());

        sc.close();
    }
}
