//Create a program, showing an example of super keyword 

   
       // Parent class
class Animal {
    String name = "Generic Animal";

    // Constructor
    Animal() {
        System.out.println("Animal constructor called");
    }

    // Method
    void makeSound() {
        System.out.println("Some generic animal sound");
    }
}

// Child class
class Dog extends Animal {
    String name = "Dog";

    // Constructor
    Dog() {
        // Calls parent class constructor
        super();
        System.out.println("Dog constructor called");
    }

    // Method overriding
    @Override
    void makeSound() {
        // Calls parent class method using super
        super.makeSound();
        System.out.println("Dog barks: Woof Woof!");
    }

    // Method to show variable access
    void displayNames() {
        System.out.println("Child class name: " + name);
        System.out.println("Parent class name: " + super.name);
    }
}

// Main class
public class SuperKeywordExample {
    public static void main(String[] args) {
        Dog d = new Dog(); // Constructor chaining demonstration

        System.out.println();
        d.displayNames();  // Accessing parent variable using super
        System.out.println();
        d.makeSound();     // Calling parent method using super
    }
}

OUTPUT

    Animal constructor called
Dog constructor called

Child class name: Dog
Parent class name: Generic Animal

Some generic animal sound
Dog barks: Woof Woof!
