PROGRAM

//Create a program, showing an example of method overriding 

   // Parent class
class Animal {
    // Method in parent class
    void makeSound() {
        System.out.println("Animal makes a sound");
    }
}

// Child class
class Dog extends Animal {
    // Overriding the parent class method
    @Override
    void makeSound() {
        System.out.println("Dog barks: Woof Woof!");
    }
}

// Main class
public class MethodOverridingExample {
    public static void main(String[] args) {
        // Creating object of parent class
        Animal a = new Animal();
        a.makeSound();  // Calls parent method

        // Creating object of child class
        Dog d = new Dog();
        d.makeSound();  // Calls overridden method

        // Using parent reference for child object (Polymorphism)
        Animal obj = new Dog();
        obj.makeSound();  // Calls Dog's version (runtime polymorphism)
    }
}
