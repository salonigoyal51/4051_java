//create an abstract class Animal with abstract methods eat() and sleep(). 

        // Abstract base class
abstract class Animal {
    String name;

    // Constructor
    Animal(String name) {
        this.name = name;
    }

    // Abstract methods (no body)
    abstract void eat();
    abstract void sleep();

    // Concrete method
    void display() {
        System.out.println("Animal Name: " + name);
    }
}

// Subclass: Lion
class Lion extends Animal {
    Lion(String name) {
        super(name);
    }

    @Override
    void eat() {
        System.out.println(name + " eats meat and hunts other animals.");
    }

    @Override
    void sleep() {
        System.out.println(name + " sleeps for about 20 hours a day in the shade.");
    }
}

// Subclass: Tiger
class Tiger extends Animal {
    Tiger(String name) {
        super(name);
    }

    @Override
    void eat() {
        System.out.println(name + " eats deer and wild boar in the forest.");
    }

    @Override
    void sleep() {
        System.out.println(name + " sleeps for around 16 hours a day in dense forest areas.");
    }
}

// Subclass: Deer
class Deer extends Animal {
    Deer(String name) {
        super(name);
    }

    @Override
    void eat() {
        System.out.println(name + " grazes on grass, leaves, and fruits.");
    }

    @Override
    void sleep() {
        System.out.println(name + " sleeps lightly for short periods to stay alert for predators.");
    }
}

// Main class
public class AbstractAnimalExample {
    public static void main(String[] args) {
        // Create objects for each subclass
        Animal lion = new Lion("Lion");
        Animal tiger = new Tiger("Tiger");
        Animal deer = new Deer("Deer");

        // Display and demonstrate behavior
        System.out.println("---- Lion Behavior ----");
        lion.display();
        lion.eat();
        lion.sleep();

        System.out.println("\n---- Tiger Behavior ----");
        tiger.display();
        tiger.eat();
        tiger.sleep();

        System.out.println("\n---- Deer Behavior ----");
        deer.display();
        deer.eat();
        deer.sleep();
    }
}
