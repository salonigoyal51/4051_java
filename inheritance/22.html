//create an abstract class Person with abstract methods eat() and exercise(). 

   // Abstract base class
abstract class Person {
    String name;

    // Constructor
    Person(String name) {
        this.name = name;
    }

    // Abstract methods
    abstract void eat();
    abstract void exercise();

    // Concrete method
    void displayInfo() {
        System.out.println("Person Name: " + name);
    }
}

// Subclass: Athlete
class Athlete extends Person {

    Athlete(String name) {
        super(name);
    }

    @Override
    void eat() {
        System.out.println(name + " eats a healthy diet full of proteins, fruits, and vegetables.");
    }

    @Override
    void exercise() {
        System.out.println(name + " exercises daily with running, swimming, and strength training.");
    }
}

// Subclass: LazyPerson
class LazyPerson extends Person {

    LazyPerson(String name) {
        super(name);
    }

    @Override
    void eat() {
        System.out.println(name + " enjoys eating fast food, snacks, and sugary drinks.");
    }

    @Override
    void exercise() {
        System.out.println(name + " rarely exercises and prefers watching TV or sleeping.");
    }
}

// Main class
public class AbstractPersonExample {
    public static void main(String[] args) {
        // Create Athlete object
        Person athlete = new Athlete("Rahul");
        System.out.println("---- Athlete Details ----");
        athlete.displayInfo();
        athlete.eat();
        athlete.exercise();

        System.out.println();

        // Create LazyPerson object
        Person lazy = new LazyPerson("Amit");
        System.out.println("---- Lazy Person Details ----");
        lazy.displayInfo();
        lazy.eat();
        lazy.exercise();
    }
}
