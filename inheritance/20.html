//create an abstract class Shape3D with abstract methods calculateVolume() and calculateSurfaceArea(). 

    // Abstract base class
abstract class Shape3D {
    String shapeName;

    // Constructor
    Shape3D(String shapeName) {
        this.shapeName = shapeName;
    }

    // Abstract methods
    abstract double calculateVolume();
    abstract double calculateSurfaceArea();

    // Concrete method
    void displayShape() {
        System.out.println("Shape: " + shapeName);
    }
}

// Subclass: Sphere
class Sphere extends Shape3D {
    double radius;

    Sphere(double radius) {
        super("Sphere");
        this.radius = radius;
    }

    @Override
    double calculateVolume() {
        return (4.0 / 3.0) * Math.PI * Math.pow(radius, 3);
    }

    @Override
    double calculateSurfaceArea() {
        return 4 * Math.PI * Math.pow(radius, 2);
    }

    void displayDetails() {
        displayShape();
        System.out.println("Radius: " + radius);
        System.out.println("Volume: " + String.format("%.2f", calculateVolume()));
        System.out.println("Surface Area: " + String.format("%.2f", calculateSurfaceArea()));
    }
}

// Subclass: Cube
class Cube extends Shape3D {
    double side;

    Cube(double side) {
        super("Cube");
        this.side = side;
    }

    @Override
    double calculateVolume() {
        return Math.pow(side, 3);
    }

    @Override
    double calculateSurfaceArea() {
        return 6 * Math.pow(side, 2);
    }

    void displayDetails() {
        displayShape();
        System.out.println("Side Length: " + side);
        System.out.println("Volume: " + String.format("%.2f", calculateVolume()));
        System.out.println("Surface Area: " + String.format("%.2f", calculateSurfaceArea()));
    }
}

// Main class
public class AbstractShape3DExample {
    public static void main(String[] args) {
        // Create objects of Sphere and Cube
        Sphere sphere = new Sphere(5.5);
        Cube cube = new Cube(4.0);

        // Display details for Sphere
        System.out.println("---- Sphere Details ----");
        sphere.displayDetails();

        System.out.println();

        // Display details for Cube
        System.out.println("---- Cube Details ----");
        cube.displayDetails();
    }
}
