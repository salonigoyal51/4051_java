//creates a class hierarchy for employees of a company. The base class should be Employee, with subclasses Manager, Developer, and Programmer. 

      // Base class
class Employee {
    String name;
    String address;
    double salary;
    String jobTitle;

    // Constructor
    Employee(String name, String address, double salary, String jobTitle) {
        this.name = name;
        this.address = address;
        this.salary = salary;
        this.jobTitle = jobTitle;
    }

    // Method to calculate bonus
    double calculateBonus() {
        System.out.println("Bonus calculation not defined for generic employee.");
        return 0.0;
    }

    // Method to generate performance report
    void generatePerformanceReport() {
        System.out.println("Performance report for " + name + ":");
        System.out.println("General employee performance evaluation.");
    }

    // Method to manage projects (default behavior)
    void manageProject() {
        System.out.println(name + " is assigned to general company projects.");
    }

    // Method to display employee details
    void displayInfo() {
        System.out.println("\n--- Employee Details ---");
        System.out.println("Name: " + name);
        System.out.println("Address: " + address);
        System.out.println("Job Title: " + jobTitle);
        System.out.println("Salary: ₹" + salary);
    }
}

// Subclass: Manager
class Manager extends Employee {
    int teamSize;

    Manager(String name, String address, double salary, String jobTitle, int teamSize) {
        super(name, address, salary, jobTitle);
        this.teamSize = teamSize;
    }

    @Override
    double calculateBonus() {
        // Managers get 20% of salary as bonus
        return salary * 0.20;
    }

    @Override
    void generatePerformanceReport() {
        System.out.println("Manager " + name + " oversees " + teamSize + " team members.");
        System.out.println("Leadership and strategic performance evaluation report generated.");
    }

    @Override
    void manageProject() {
        System.out.println(name + " is managing multiple projects and ensuring team productivity.");
    }

    @Override
    void displayInfo() {
        super.displayInfo();
        System.out.println("Team Size: " + teamSize);
    }
}

// Subclass: Developer
class Developer extends Employee {
    String programmingLanguage;

    Developer(String name, String address, double salary, String jobTitle, String programmingLanguage) {
        super(name, address, salary, jobTitle);
        this.programmingLanguage = programmingLanguage;
    }

    @Override
    double calculateBonus() {
        // Developers get 15% of salary as bonus
        return salary * 0.15;
    }

    @Override
    void generatePerformanceReport() {
        System.out.println("Developer " + name + " shows strong coding and debugging performance.");
        System.out.println("Performance report generated for language: " + programmingLanguage);
    }

    @Override
    void manageProject() {
        System.out.println(name + " is developing modules using " + programmingLanguage + ".");
    }

    @Override
    void displayInfo() {
        super.displayInfo();
        System.out.println("Programming Language: " + programmingLanguage);
    }
}

// Subclass: Programmer
class Programmer extends Employee {
    int linesOfCodePerDay;

    Programmer(String name, String address, double salary, String jobTitle, int linesOfCodePerDay) {
        super(name, address, salary, jobTitle);
        this.linesOfCodePerDay = linesOfCodePerDay;
    }

    @Override
    double calculateBonus() {
        // Programmers get 10% of salary as bonus
        return salary * 0.10;
    }

    @Override
    void generatePerformanceReport() {
        System.out.println("Programmer " + name + " writes approximately " + linesOfCodePerDay + " lines of code daily.");
        System.out.println("Efficiency and accuracy report generated.");
    }

    @Override
    void manageProject() {
        System.out.println(name + " contributes to project development and code maintenance.");
    }

    @Override
    void displayInfo() {
        super.displayInfo();
        System.out.println("Average Lines of Code per Day: " + linesOfCodePerDay);
    }
}

// Main class
public class EmployeeHierarchyExample {
    public static void main(String[] args) {
        // Create objects for each type of employee
        Manager manager = new Manager("Alice Johnson", "Mumbai", 120000, "Project Manager", 10);
        Developer developer = new Developer("Ravi Kumar", "Delhi", 80000, "Java Developer", "Java");
        Programmer programmer = new Programmer("Sneha Sharma", "Pune", 60000, "Software Programmer", 500);

        // Display details and operations
        manager.displayInfo();
        manager.manageProject();
        manager.generatePerformanceReport();
        System.out.println("Bonus: ₹" + manager.calculateBonus());

        developer.displayInfo();
        developer.manageProject();
        developer.generatePerformanceReport();
        System.out.println("Bonus: ₹" + developer.calculateBonus());

        programmer.displayInfo();
        programmer.manageProject();
        programmer.generatePerformanceReport();
        System.out.println("Bonus: ₹" + programmer.calculateBonus());
    }
}
