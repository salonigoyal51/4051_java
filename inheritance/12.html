//Create a program, showing an example of super function 

     
      // Parent class
class Vehicle {
    String brand;
    int year;

    // Parameterized constructor
    Vehicle(String brand, int year) {
        this.brand = brand;
        this.year = year;
        System.out.println("Vehicle constructor called");
    }

    void displayInfo() {
        System.out.println("Brand: " + brand);
        System.out.println("Year: " + year);
    }
}

// Child class
class Car extends Vehicle {
    String model;

    // Parameterized constructor using super()
    Car(String brand, int year, String model) {
        // Calls parent class constructor
        super(brand, year);
        this.model = model;
        System.out.println("Car constructor called");
    }

    void showDetails() {
        // Access parent class method
        super.displayInfo();
        System.out.println("Model: " + model);
    }
}

// Main class
public class SuperFunctionExample {
    public static void main(String[] args) {
        // Create object of subclass
        Car car = new Car("Toyota", 2023, "Camry");

        System.out.println();
        car.showDetails();
    }
}

OUTPUT
