//Print the elements of an array present on even position. 

   public class EvenPositionElements {
    public static void main(String[] args) {
        // You can change the array elements here
        int[] arr = {10, 20, 30, 40, 50, 60};

        System.out.println("Elements of the array present on even positions:");

        // Loop through even positions (2nd, 4th, 6th, etc.)
        for (int i = 1; i < arr.length; i += 2) { // index 1 = 2nd position
            System.out.print(arr[i] + " ");
        }
    }
}

OUTPUT

    Elements of the array present on even positions:
   20 40 60
