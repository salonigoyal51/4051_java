//create a method that takes an integer as a parameter and throws an exception if the number is odd. 

       
     // Custom exception class
class OddNumberException extends Exception {
    OddNumberException(String message) {
        super(message);
    }
}

public class OddNumberChecker {

    // Method that checks if number is odd and throws exception
    static void checkEven(int number) throws OddNumberException {
        if (number % 2 != 0) {
            throw new OddNumberException("Number " + number + " is odd!");
        } else {
            System.out.println("Number " + number + " is even.");
        }
    }

    public static void main(String[] args) {
        int[] numbers = {4, 7, 10, 13};

        for (int num : numbers) {
            try {
                checkEven(num);
            } catch (OddNumberException e) {
                System.out.println("Exception caught: " + e.getMessage());
            }
        }

        System.out.println("Program continues after checking numbers.");
    }
}

OUTPUT

Number 4 is even.
Exception caught: Number 7 is odd!
Number 10 is even.
Exception caught: Number 13 is odd!
Program continues after checking numbers.
