//read a list of integers and throws an exception if any numbers are duplicates 

 import java.util.*;
import java.io.*;

// Custom exception class
class DuplicateNumberException extends Exception {
    DuplicateNumberException(String message) {
        super(message);
    }
}

public class DuplicateNumberChecker {

    // Method to check for duplicates in the list
    static void checkDuplicates(List numbers) throws DuplicateNumberException {
        Set uniqueNumbers = new HashSet<>();

        for (int num : numbers) {
            if (!uniqueNumbers.add(num)) { // add() returns false if element already exists
                throw new DuplicateNumberException("Duplicate number found: " + num);
            }
        }
        System.out.println("No duplicates found. All numbers are unique.");
    }

    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));

        System.out.print("Enter integers separated by space: ");
        String input = br.readLine();

        String[] tokens = input.trim().split("\\s+");
        List numbers = new ArrayList<>();

        for (String token : tokens) {
            try {
                numbers.add(Integer.parseInt(token));
            } catch (NumberFormatException e) {
                System.out.println("Skipping invalid input: " + token);
            }
        }

        try {
            checkDuplicates(numbers);
        } catch (DuplicateNumberException e) {
            System.out.println("Exception caught: " + e.getMessage());
        }

        System.out.println("Program finished.");
    }
}
