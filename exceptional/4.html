//program that reads a list of numbers from a file and throws an exception if any of the numbers are positive 

        
    import java.io.*;

// Custom exception class
class PositiveNumberException extends Exception {
    PositiveNumberException(String message) {
        super(message);
    }
}

public class NegativeNumberFileReader {

    // Method to read numbers from a file and check for positive numbers
    static void readNumbers(String filename) throws IOException, PositiveNumberException {
        BufferedReader br = new BufferedReader(new FileReader(filename));
        String line;
        int lineNumber = 1;

        while ((line = br.readLine()) != null) {
            try {
                int number = Integer.parseInt(line.trim());

                if (number > 0) {
                    throw new PositiveNumberException(
                        "Positive number found at line " + lineNumber + ": " + number
                    );
                }

            } catch (NumberFormatException e) {
                System.out.println("Skipping invalid number at line " + lineNumber + ": " + line);
            }

            lineNumber++;
        }

        br.close();
    }

    public static void main(String[] args) {
        String filename = "numbers.txt"; // Replace with your file path

        try {
            readNumbers(filename);
            System.out.println("All numbers are zero or negative.");
        } catch (PositiveNumberException e) {
            System.out.println("Exception caught: " + e.getMessage());
        } catch (IOException e) {
            System.out.println("IOException caught: " + e.getMessage());
        }

        System.out.println("Program finished.");
    }
}
