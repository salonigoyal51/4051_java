//create a method that reads a file and throws an exception if the file is not found 

    import java.io.*;

public class FileReadExample {

    // Method to read a file
    static void readFile(String filename) throws FileNotFoundException, IOException {
        FileReader fr = new FileReader(filename); // Throws FileNotFoundException if file doesn't exist
        BufferedReader br = new BufferedReader(fr);

        String line;
        System.out.println("Contents of the file:");
        while ((line = br.readLine()) != null) {
            System.out.println(line);
        }

        br.close();
        fr.close();
    }

    public static void main(String[] args) {
        String filename = "example.txt"; // Change to your file name

        try {
            readFile(filename);
        } catch (FileNotFoundException e) {
            System.out.println("Exception caught: File not found - " + e.getMessage());
        } catch (IOException e) {
            System.out.println("Exception caught: IO error - " + e.getMessage());
        }

        System.out.println("Program continues after file reading attempt.");
    }
}
