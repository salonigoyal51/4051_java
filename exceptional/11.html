//program to show an example of throws Keyword 

    import java.io.*;

public class ThrowsExample {

    // Method that declares it throws IOException
    static void readFile(String filename) throws IOException {
        FileReader fr = new FileReader(filename); // Might throw FileNotFoundException
        BufferedReader br = new BufferedReader(fr);

        System.out.println("Contents of the file:");
        String line;
        while ((line = br.readLine()) != null) {
            System.out.println(line);
        }

        br.close();
        fr.close();
    }

    public static void main(String[] args) {
        String filename = "example.txt"; // Replace with your file

        try {
            readFile(filename); // Call method that throws IOException
        } catch (IOException e) {
            System.out.println("Exception caught: " + e.getMessage());
        }

        System.out.println("Program continues after reading the file.");
    }
}
