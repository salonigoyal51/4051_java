//Create a List Box to Select Multiple Items and Display it in the Frame 

 import java.awt.*;
import java.awt.event.*;

public class MultiSelectListExample extends Frame implements ActionListener {

    List listBox;
    Button showButton;
    Label label;

    public MultiSelectListExample() {
        // Frame title
        super("Multiple Selection List Box");

        // Layout
        setLayout(new FlowLayout());

        // Create List box with multiple selection enabled
        listBox = new List(5, true); // 5 visible rows, multiple selection allowed
        listBox.add("Java");
        listBox.add("Python");
        listBox.add("C++");
        listBox.add("JavaScript");
        listBox.add("C#");
        listBox.add("Ruby");
        listBox.add("Go");
        add(listBox);

        // Button to display selected items
        showButton = new Button("Show Selected");
        add(showButton);

        // Label to display selection
        label = new Label("Selected Items will appear here");
        add(label);

        // Register event listener
        showButton.addActionListener(this);

        // Frame properties
        setSize(400, 300);
        setVisible(true);

        // Window closing
        addWindowListener(new WindowAdapter() {
            public void windowClosing(WindowEvent e) {
                dispose();
            }
        });
    }

    // Handle button click
    public void actionPerformed(ActionEvent e) {
        String[] selectedItems = listBox.getSelectedItems();
        if (selectedItems.length == 0) {
            label.setText("No item selected");
        } else {
            String items = String.join(", ", selectedItems);
            label.setText("Selected Items: " + items);
        }
    }

    public static void main(String[] args) {
        new MultiSelectListExample();
    }
}
