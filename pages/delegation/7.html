//Program to Get and Set State and Get Label of a Check Box 

  import java.awt.*;
import java.awt.event.*;

public class CheckboxExample extends Frame implements ItemListener {

    Label label;
    Checkbox checkbox;

    public CheckboxExample() {
        // Frame title
        super("Checkbox Example");

        // Layout
        setLayout(new FlowLayout());

        // Create a Checkbox
        checkbox = new Checkbox("Subscribe to Newsletter");
        add(checkbox);

        // Label to display checkbox state and label
        label = new Label("Checkbox State: " + checkbox.getState() + ", Label: " + checkbox.getLabel());
        add(label);

        // Register ItemListener (delegation event model)
        checkbox.addItemListener(this);

        // Frame properties
        setSize(400, 150);
        setVisible(true);

        // Window closing
        addWindowListener(new WindowAdapter() {
            public void windowClosing(WindowEvent e) {
                dispose();
            }
        });
    }

    // Handle checkbox state changes
    public void itemStateChanged(ItemEvent e) {
        // Get state and label
        boolean state = checkbox.getState();
        String text = checkbox.getLabel();

        // Update label
        label.setText("Checkbox State: " + state + ", Label: " + text);
    }

    public static void main(String[] args) {
        new CheckboxExample();
    }
}

OUTPUT

After clicking the checkbox:

Checkbox State: true, Label: Subscribe to Newsletter


After clicking again (uncheck):

[ ] Subscribe to Newsletter
Checkbox State: false, Label: Subscribe to Newsletter
