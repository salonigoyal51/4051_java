//Create Text Area and Password Field 

 import java.awt.*;
import java.awt.event.*;

public class TextAreaPasswordExample extends Frame implements ActionListener {

    Label label;
    TextArea textArea;
    TextField passwordField;
    Button submitButton;

    public TextAreaPasswordExample() {
        // Set frame title
        super("TextArea and PasswordField Example");

        // Set layout
        setLayout(new FlowLayout());

        // Label
        label = new Label("Enter text and password, then press Enter or click Submit:");
        add(label);

        // TextArea
        textArea = new TextArea("", 5, 30, TextArea.SCROLLBARS_VERTICAL_ONLY);
        add(textArea);
        textArea.addActionListener(this); // TextArea does not generate ActionEvent on Enter, handled with button

        // PasswordField
        passwordField = new TextField(20);
        passwordField.setEchoChar('*');
        add(passwordField);
        passwordField.addActionListener(this); // Handles Enter key

        // Submit Button
        submitButton = new Button("Submit");
        add(submitButton);
        submitButton.addActionListener(this);

        // Frame properties
        setSize(400, 300);
        setVisible(true);

        // Window closing
        addWindowListener(new WindowAdapter() {
            public void windowClosing(WindowEvent e) {
                dispose();
            }
        });
    }

    // Handle events
    public void actionPerformed(ActionEvent e) {
        String text = textArea.getText();
        String password = passwordField.getText();

        // Display in a dialog or update label
        label.setText("Text: " + text + " | Password: " + password);
    }

    public static void main(String[] args) {
        new TextAreaPasswordExample();
    }
}
