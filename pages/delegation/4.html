//Create 2 Radio Buttons and Display Selected Button Label 

 import java.awt.*;
import java.awt.event.*;

public class RadioButtonExample extends Frame implements ItemListener {

    Label label;
    Checkbox radio1, radio2;
    CheckboxGroup group;

    public RadioButtonExample() {
        // Frame title
        super("Radio Button Example");

        // Layout
        setLayout(new FlowLayout());

        // Label to display selection
        label = new Label("Select a radio button:");
        add(label);

        // Create CheckboxGroup for radio buttons
        group = new CheckboxGroup();

        // Create radio buttons
        radio1 = new Checkbox("Option 1", group, false);
        radio2 = new Checkbox("Option 2", group, false);

        // Add radio buttons to frame
        add(radio1);
        add(radio2);

        // Register ItemListener (delegation event model)
        radio1.addItemListener(this);
        radio2.addItemListener(this);

        // Frame properties
        setSize(300, 150);
        setVisible(true);

        // Window closing
        addWindowListener(new WindowAdapter() {
            public void windowClosing(WindowEvent e) {
                dispose();
            }
        });
    }

    // Handle radio button selection
    public void itemStateChanged(ItemEvent e) {
        Checkbox selected = group.getSelectedCheckbox();
        label.setText("Selected: " + selected.getLabel());
    }

    public static void main(String[] args) {
        new RadioButtonExample();
    }
}
