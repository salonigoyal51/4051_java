//Program to Change Frame Background Color as Cyan 

   import java.awt.*;
import java.awt.event.*;

public class FrameBackgroundColor extends Frame {

    public FrameBackgroundColor() {
        // Frame title
        super("Frame Background Color Example");

        // Set layout
        setLayout(new FlowLayout());

        // Set background color to Cyan
        setBackground(Color.CYAN);

        // Add a label
        Label label = new Label("This frame has a cyan background!");
        add(label);

        // Frame properties
        setSize(400, 200);
        setVisible(true);

        // Window closing
        addWindowListener(new WindowAdapter() {
            public void windowClosing(WindowEvent e) {
                dispose();
            }
        });
    }

    public static void main(String[] args) {
        new FrameBackgroundColor();
    }
}

OUTPUT

The frame appears with a cyan background and a label:

This frame has a cyan background!
