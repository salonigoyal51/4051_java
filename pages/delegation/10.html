//Program to Create and Set Border to Push Buttons 

  import java.awt.*;
import java.awt.event.*;

public class ButtonBorderExample extends Frame implements ActionListener {

    Button btn1, btn2, btn3;
    Label label;

    public ButtonBorderExample() {
        // Frame title
        super("Push Buttons with Borders");

        // Layout
        setLayout(new FlowLayout());

        // Label
        label = new Label("Click any button");
        add(label);

        // Create buttons
        btn1 = new Button("Button 1");
        btn2 = new Button("Button 2");
        btn3 = new Button("Button 3");

        // Simulate border using Panel
        Panel p1 = new Panel(new FlowLayout());
        p1.setBackground(Color.LIGHT_GRAY); // Border color
        p1.add(btn1);
        add(p1);

        Panel p2 = new Panel(new FlowLayout());
        p2.setBackground(Color.LIGHT_GRAY);
        p2.add(btn2);
        add(p2);

        Panel p3 = new Panel(new FlowLayout());
        p3.setBackground(Color.LIGHT_GRAY);
        p3.add(btn3);
        add(p3);

        // Register event listeners
        btn1.addActionListener(this);
        btn2.addActionListener(this);
        btn3.addActionListener(this);

        // Frame properties
        setSize(400, 200);
        setVisible(true);

        // Window closing
        addWindowListener(new WindowAdapter() {
            public void windowClosing(WindowEvent e) {
                dispose();
            }
        });
    }

    // Handle button clicks
    public void actionPerformed(ActionEvent e) {
        Button b = (Button) e.getSource();
        label.setText("You clicked: " + b.getLabel());
    }

    public static void main(String[] args) {
        new ButtonBorderExample();
    }
}

OUTPUT
