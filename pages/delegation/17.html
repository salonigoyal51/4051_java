//Program to Change the Applet Background Colour when Button Clicked 

  import java.applet.*;
import java.awt.*;
import java.awt.event.*;

/*

*/

public class ChangeBgColorApplet extends Applet implements ActionListener {

    Button cyanButton, pinkButton;

    public void init() {
        // Set initial background color
        setBackground(Color.LIGHT_GRAY);

        // Create buttons
        cyanButton = new Button("Cyan");
        pinkButton = new Button("Pink");

        // Add buttons to the applet
        add(cyanButton);
        add(pinkButton);

        // Register event listeners
        cyanButton.addActionListener(this);
        pinkButton.addActionListener(this);
    }

    // Handle button clicks
    public void actionPerformed(ActionEvent e) {
        if (e.getSource() == cyanButton) {
            setBackground(Color.CYAN);
        } else if (e.getSource() == pinkButton) {
            setBackground(Color.PINK);
        }
        repaint();
    }
}

OUTPUT

Initial Applet:

[Button] Cyan   [Button] Pink
Background: Light Gray


After clicking "Cyan":

Background changes to Cyan.

After clicking "Pink":

Background changes to Pink.
