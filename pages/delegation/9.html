//Program to Create and Switch between Frames using Buttons 

 import java.awt.*;
import java.awt.event.*;

public class FrameSwitchExample {

    public static void main(String[] args) {
        // Create first frame
        Frame frame1 = new Frame("Frame 1");
        frame1.setLayout(new FlowLayout());
        frame1.setSize(300, 200);

        Label label1 = new Label("This is Frame 1");
        frame1.add(label1);

        Button toFrame2 = new Button("Go to Frame 2");
        frame1.add(toFrame2);

        // Create second frame
        Frame frame2 = new Frame("Frame 2");
        frame2.setLayout(new FlowLayout());
        frame2.setSize(300, 200);

        Label label2 = new Label("This is Frame 2");
        frame2.add(label2);

        Button toFrame1 = new Button("Back to Frame 1");
        frame2.add(toFrame1);

        // Event handling using delegation model
        toFrame2.addActionListener(new ActionListener() {
            public void actionPerformed(ActionEvent e) {
                frame1.setVisible(false);
                frame2.setVisible(true);
            }
        });

        toFrame1.addActionListener(new ActionListener() {
            public void actionPerformed(ActionEvent e) {
                frame2.setVisible(false);
                frame1.setVisible(true);
            }
        });

        // Window closing
        frame1.addWindowListener(new WindowAdapter() {
            public void windowClosing(WindowEvent e) {
                frame1.dispose();
            }
        });

        frame2.addWindowListener(new WindowAdapter() {
            public void windowClosing(WindowEvent e) {
                frame2.dispose();
            }
        });

        // Show first frame
        frame1.setVisible(true);
    }
}

OUTPUT

Frame 1 initially:

This is Frame 1
[Button] Go to Frame 2


After clicking "Go to Frame 2":

This is Frame 2
[Button] Back to Frame 1


Clicking "Back to Frame 1" returns to Frame 1.
