//Program to Create Check Boxes and Radio Buttons 

import java.awt.*;
import java.awt.event.*;

public class CheckboxRadioExample extends Frame implements ItemListener {

    Label label;
    Checkbox cb1, cb2, cb3;
    Checkbox rb1, rb2, rb3;
    CheckboxGroup radioGroup;

    public CheckboxRadioExample() {
        // Frame title
        super("Checkboxes and Radio Buttons Example");

        // Layout
        setLayout(new FlowLayout());

        // Checkboxes
        cb1 = new Checkbox("Java");
        cb2 = new Checkbox("Python");
        cb3 = new Checkbox("C++");
        add(cb1);
        add(cb2);
        add(cb3);

        // Radio buttons (CheckboxGroup)
        radioGroup = new CheckboxGroup();
        rb1 = new Checkbox("Male", radioGroup, false);
        rb2 = new Checkbox("Female", radioGroup, false);
        rb3 = new Checkbox("Other", radioGroup, false);
        add(rb1);
        add(rb2);
        add(rb3);

        // Label to show selected states
        label = new Label("Select checkboxes and a radio button.");
        add(label);

        // Register ItemListener
        cb1.addItemListener(this);
        cb2.addItemListener(this);
        cb3.addItemListener(this);
        rb1.addItemListener(this);
        rb2.addItemListener(this);
        rb3.addItemListener(this);

        // Frame properties
        setSize(400, 250);
        setVisible(true);

        // Window closing
        addWindowListener(new WindowAdapter() {
            public void windowClosing(WindowEvent e) {
                dispose();
            }
        });
    }

    // Handle item events
    public void itemStateChanged(ItemEvent e) {
        // Checkboxes selected
        String checkboxes = "Checkboxes selected: ";
        if (cb1.getState()) checkboxes += cb1.getLabel() + " ";
        if (cb2.getState()) checkboxes += cb2.getLabel() + " ";
        if (cb3.getState()) checkboxes += cb3.getLabel() + " ";

        // Selected radio button
        Checkbox selectedRadio = radioGroup.getSelectedCheckbox();
        String radio = " | Selected radio button: ";
        radio += (selectedRadio != null) ? selectedRadio.getLabel() : "None";

        // Update label
        label.setText(checkboxes + radio);
    }

    public static void main(String[] args) {
        new CheckboxRadioExample();
    }
}
