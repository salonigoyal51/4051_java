//Program to Handle KeyBoardEvent 

   import java.awt.*;
import java.awt.event.*;

public class KeyboardEventExample extends Frame implements KeyListener {

    Label label;
    TextField textField;

    public KeyboardEventExample() {
        // Frame title
        super("Keyboard Event Example");

        // Layout
        setLayout(new FlowLayout());

        // Label
        label = new Label("Press any key in the text field:");
        add(label);

        // TextField to capture key events
        textField = new TextField(20);
        add(textField);

        // Register KeyListener (delegation event model)
        textField.addKeyListener(this);

        // Frame properties
        setSize(400, 150);
        setVisible(true);

        // Window closing
        addWindowListener(new WindowAdapter() {
            public void windowClosing(WindowEvent e) {
                dispose();
            }
        });
    }

    // KeyListener methods
    public void keyPressed(KeyEvent e) {
        label.setText("Key Pressed: " + e.getKeyChar());
    }

    public void keyReleased(KeyEvent e) {
        // Optional: Can display released key
    }

    public void keyTyped(KeyEvent e) {
        // Optional: Can display typed key
    }

    public static void main(String[] args) {
        new KeyboardEventExample();
    }
}

OUTPUT

   Initial Frame:

Press any key in the text field:
[TextField] (empty)

Key Pressed: 5
