//Display Some Text in the Frame using Labels 

  import java.awt.*;
import java.awt.event.*;

public class LabelEventExample extends Frame implements ActionListener {

    Label label;
    Button button;

    public LabelEventExample() {
        // Set frame title
        super("Delegation Event Example");

        // Set layout
        setLayout(new FlowLayout());

        // Create Label
        label = new Label("Click the button to display text.");
        add(label);

        // Create Button
        button = new Button("Click Me");
        add(button);

        // Register event listener (delegation model)
        button.addActionListener(this);

        // Frame properties
        setSize(400, 150);
        setVisible(true);

        // Window closing event
        addWindowListener(new WindowAdapter() {
            public void windowClosing(WindowEvent e) {
                dispose();
            }
        });
    }

    // Handle button click
    public void actionPerformed(ActionEvent e) {
        label.setText("Hello! This text is displayed in the Label.");
    }

    public static void main(String[] args) {
        new LabelEventExample();
    }
}
