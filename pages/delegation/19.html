//Program to Validate the TextField for only entering numbers using KeyListener 

 import java.awt.*;
import java.awt.event.*;

public class NumberOnlyTextField extends Frame implements KeyListener {

    Label label;
    TextField textField;

    public NumberOnlyTextField() {
        // Frame title
        super("Number Only TextField Validation");

        // Layout
        setLayout(new FlowLayout());

        // Label
        label = new Label("Enter numbers only:");
        add(label);

        // TextField
        textField = new TextField(20);
        add(textField);

        // Register KeyListener
        textField.addKeyListener(this);

        // Frame properties
        setSize(400, 150);
        setVisible(true);

        // Window closing
        addWindowListener(new WindowAdapter() {
            public void windowClosing(WindowEvent e) {
                dispose();
            }
        });
    }

    // KeyListener methods
    public void keyTyped(KeyEvent e) {
        char ch = e.getKeyChar();
        // Allow digits and control keys (like backspace)
        if (!Character.isDigit(ch) && ch != KeyEvent.VK_BACK_SPACE) {
            e.consume(); // Ignore the character
        }
    }

    public void keyPressed(KeyEvent e) {
        // Not used
    }

    public void keyReleased(KeyEvent e) {
        // Not used
    }

    public static void main(String[] args) {
        new NumberOnlyTextField();
    }
}
