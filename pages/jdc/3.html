//Insert the row in the table in the database from Java API. 

  import java.sql.Connection;
import java.sql.DriverManager;
import java.sql.PreparedStatement;
import java.sql.SQLException;

public class InsertRowExample {
    public static void main(String[] args) {
        String url = "jdbc:mysql://localhost:3306/testdb"; // your DB name
        String user = "root"; // your MySQL username
        String password = "1234"; // your MySQL password

        Connection conn = null;
        PreparedStatement pstmt = null;

        try {
            // 1. Load MySQL JDBC Driver
            Class.forName("com.mysql.cj.jdbc.Driver");
            System.out.println("Driver Loaded Successfully!");

            // 2. Establish the connection
            conn = DriverManager.getConnection(url, user, password);
            System.out.println("âœ… Connection Established Successfully!");

            // 3. Create SQL INSERT query
            String insertSQL = "INSERT INTO students (name, age, course) VALUES (?, ?, ?)";

            // 4. Create PreparedStatement
            pstmt = conn.prepareStatement(insertSQL);

            // 5. Set parameter values
            pstmt.setString(1, "Anuradha");
            pstmt.setInt(2, 22);
            pstmt.setString(3, "Computer Science");

            // 6. Execute the INSERT command
            int rowsInserted = pstmt.executeUpdate();

            if (rowsInserted > 0) {
                System.out.println("ğŸ‰ A new student record was inserted successfully!");
            }

        } catch (ClassNotFoundException e) {
            System.out.println("âŒ MySQL JDBC Driver not found!");
            e.printStackTrace();
        } catch (SQLException e) {
            System.out.println("âŒ Database operation failed!");
            e.printStackTrace();
        } finally {
            // 7. Close resources
            try {
                if (pstmt != null) pstmt.close();
                if (conn != null) conn.close();
                System.out.println("ğŸ”’ Connection Closed.");
            } catch (SQLException e) {
                e.printStackTrace();
            }
        }
    }
}
