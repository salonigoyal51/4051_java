// Delete the row in the table in the database from Java API. 

  import java.sql.Connection;
import java.sql.DriverManager;
import java.sql.PreparedStatement;
import java.sql.SQLException;

public class DeleteRowExample {
    public static void main(String[] args) {
        String url = "jdbc:mysql://localhost:3306/testdb"; // your database name
        String user = "root"; // your MySQL username
        String password = "1234"; // your MySQL password

        Connection conn = null;
        PreparedStatement pstmt = null;

        try {
            // 1. Load the MySQL JDBC Driver
            Class.forName("com.mysql.cj.jdbc.Driver");
            System.out.println("Driver Loaded Successfully!");

            // 2. Establish the connection
            conn = DriverManager.getConnection(url, user, password);
            System.out.println("‚úÖ Connection Established Successfully!");

            // 3. Create SQL DELETE query
            String deleteSQL = "DELETE FROM students WHERE name = ?";

            // 4. Create PreparedStatement
            pstmt = conn.prepareStatement(deleteSQL);

            // 5. Set the parameter value
            pstmt.setString(1, "Anuradha"); // deletes the record with name 'Anuradha'

            // 6. Execute the DELETE command
            int rowsDeleted = pstmt.executeUpdate();

            if (rowsDeleted > 0) {
                System.out.println("üóëÔ∏è Record deleted successfully!");
            } else {
                System.out.println("‚ö†Ô∏è No matching record found to delete!");
            }

        } catch (ClassNotFoundException e) {
            System.out.println("‚ùå MySQL JDBC Driver not found!");
            e.printStackTrace();
        } catch (SQLException e) {
            System.out.println("‚ùå Database operation failed!");
            e.printStackTrace();
        } finally {
            // 7. Close resources
            try {
                if (pstmt != null) pstmt.close();
                if (conn != null) conn.close();
                System.out.println("üîí Connection Closed.");
            } catch (SQLException e) {
                e.printStackTrace();
            }
        }
    }
}
