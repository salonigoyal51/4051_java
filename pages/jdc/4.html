//Update the row in the table in the database from Java API. 

 import java.sql.Connection;
import java.sql.DriverManager;
import java.sql.PreparedStatement;
import java.sql.SQLException;

public class UpdateRowExample {
    public static void main(String[] args) {
        String url = "jdbc:mysql://localhost:3306/testdb"; // your database name
        String user = "root"; // your MySQL username
        String password = "1234"; // your MySQL password

        Connection conn = null;
        PreparedStatement pstmt = null;

        try {
            // 1. Load the MySQL JDBC driver
            Class.forName("com.mysql.cj.jdbc.Driver");
            System.out.println("Driver Loaded Successfully!");

            // 2. Establish the connection
            conn = DriverManager.getConnection(url, user, password);
            System.out.println("‚úÖ Connection Established Successfully!");

            // 3. Create SQL UPDATE query
            String updateSQL = "UPDATE students SET age = ?, course = ? WHERE name = ?";

            // 4. Create PreparedStatement
            pstmt = conn.prepareStatement(updateSQL);

            // 5. Set the parameters
            pstmt.setInt(1, 23); // new age
            pstmt.setString(2, "Data Science"); // new course
            pstmt.setString(3, "Anuradha"); // which record to update (WHERE name = 'Anuradha')

            // 6. Execute the UPDATE command
            int rowsUpdated = pstmt.executeUpdate();

            if (rowsUpdated > 0) {
                System.out.println("üéâ Record updated successfully!");
            } else {
                System.out.println("‚ö†Ô∏è No matching record found to update!");
            }

        } catch (ClassNotFoundException e) {
            System.out.println("‚ùå MySQL JDBC Driver not found!");
            e.printStackTrace();
        } catch (SQLException e) {
            System.out.println("‚ùå Database operation failed!");
            e.printStackTrace();
        } finally {
            // 7. Close resources
            try {
                if (pstmt != null) pstmt.close();
                if (conn != null) conn.close();
                System.out.println("üîí Connection Closed.");
            } catch (SQLException e) {
                e.printStackTrace();
            }
        }
    }
}
