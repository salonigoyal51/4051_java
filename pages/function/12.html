//Write a Java method that accepts three integers and returns true if one is the middle point between the other two integers, otherwise false 

 import java.util.Scanner;

public class MiddlePointChecker {

    // Method to check if one number is the middle point between the other two
    static boolean isMiddlePoint(int a, int b, int c) {
        // Check if b is the middle between a and c
        if ((a < b && b < c) || (c < b && b < a)) return true;

        // Check if a is the middle between b and c
        if ((b < a && a < c) || (c < a && a < b)) return true;

        // Check if c is the middle between a and b
        if ((a < c && c < b) || (b < c && c < a)) return true;

        return false;
    }

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        System.out.print("Enter first integer: ");
        int num1 = sc.nextInt();

        System.out.print("Enter second integer: ");
        int num2 = sc.nextInt();

        System.out.print("Enter third integer: ");
        int num3 = sc.nextInt();

        if (isMiddlePoint(num1, num2, num3)) {
            System.out.println("One of the numbers is the middle point between the other two.");
        } else {
            System.out.println("None of the numbers is the middle point.");
        }

        sc.close();
    }
}
