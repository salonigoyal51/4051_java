//count the number of digits in an integer with the value 2. The integer may be assumed to be non-negative 

  import java.util.Scanner;

public class CountDigitTwo {

    // Method to count the number of digits equal to 2
    static int countTwos(int number) {
        int count = 0;

        // Handle the case when number is 0
        if (number == 0) {
            return 0;
        }

        while (number > 0) {
            int digit = number % 10;
            if (digit == 2) {
                count++;
            }
            number = number / 10;
        }

        return count;
    }

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        System.out.print("Enter a non-negative integer: ");
        int num = sc.nextInt();

        int twosCount = countTwos(num);
        System.out.println("Number of digits equal to 2: " + twosCount);

        sc.close();
    }
}
