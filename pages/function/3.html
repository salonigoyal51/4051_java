//display the middle character of a string. 

    import java.util.Scanner;

public class MiddleCharacter {

    // Method to display middle character(s)
    static void displayMiddle(String str) {
        int length = str.length();
        if (length == 0) {
            System.out.println("The string is empty.");
            return;
        }

        if (length % 2 == 0) {
            // Even length → one middle character
            int midIndex = length / 2;
            System.out.println("Middle character: " + str.charAt(midIndex));
        } else {
            // Odd length → two middle characters
            int midIndex = length / 2;
            System.out.println("Middle characters: " + str.charAt(midIndex) + str.charAt(midIndex + 1));
        }
    }

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        System.out.print("Enter a string: ");
        String input = sc.nextLine();

        displayMiddle(input);

        sc.close();
    }
}
