//count all the words in a string 

    import java.util.Scanner;

public class WordCounter {

    // Method to count words in a string
    static int countWords(String str) {
        if (str == null || str.isEmpty()) {
            return 0; // No words in empty string
        }

        // Split string by spaces (one or more spaces) and count
        String[] words = str.trim().split("\\s+");
        return words.length;
    }

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        System.out.print("Enter a string: ");
        String input = sc.nextLine();

        int wordCount = countWords(input);
        System.out.println("The number of words in the string: " + wordCount);

        sc.close();
    }
}
