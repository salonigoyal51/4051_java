//checks whether all the characters in a given string are vowels (a, e,i,o,u) or not. Return true if each character in the string is a vowel, otherwise return false. 

   
     import java.util.Scanner;

public class AllVowelsChecker {

    // Method to check if all characters are vowels
    static boolean allVowels(String str) {
        if (str == null || str.isEmpty()) {
            return false; // Empty string considered not all vowels
        }

        str = str.toLowerCase(); // Convert to lowercase for simplicity

        for (char c : str.toCharArray()) {
            if (c != 'a' && c != 'e' && c != 'i' && c != 'o' && c != 'u') {
                return false; // Found a non-vowel character
            }
        }

        return true; // All characters are vowels
    }

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        System.out.print("Enter a string: ");
        String input = sc.nextLine();

        if (allVowels(input)) {
            System.out.println("All characters in the string are vowels.");
        } else {
            System.out.println("Not all characters in the string are vowels.");
        }

        sc.close();
    }
}
