//Write a Java method to print characters between two characters (i.e. A to P). 

        
     import java.util.Scanner;

public class PrintCharacters {

    // Method to print characters between two characters
    static void printCharsBetween(char start, char end) {
        if (start > end) {
            System.out.println("Invalid input: start character comes after end character.");
            return;
        }

        System.out.print("Characters from " + start + " to " + end + ": ");
        for (char c = start; c <= end; c++) {
            System.out.print(c + " ");
        }
        System.out.println();
    }

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        System.out.print("Enter start character: ");
        char startChar = sc.next().charAt(0);

        System.out.print("Enter end character: ");
        char endChar = sc.next().charAt(0);

        printCharsBetween(startChar, endChar);

        sc.close();
    }
}
