// Enter the boolean from user using Scanner class 

   import java.util.Scanner;

public class ReadBooleanExample {
    public static void main(String[] args) {
        // Create Scanner object for user input
        Scanner sc = new Scanner(System.in);

        // Prompt the user for a boolean value
        System.out.print("Enter a boolean value (true/false): ");
        boolean value = sc.nextBoolean();  // Read boolean input

        // Display the entered boolean value
        System.out.println("You entered: " + value);

        // Close the scanner
        sc.close();
    }
}

OUTPUT

Enter a boolean value (true/false): true
You entered: true
