PROGRAM

// Get all characters from the file, count number of lines, words, characters and display on the screen 

import java.io.*;

public class FileReadCount {
    public static void main(String[] args) {
        // File name (make sure this file exists in your project folder)
        String fileName = "input.txt";

        int lines = 0, words = 0, chars = 0;

        try {
            // Create BufferedReader to read text efficiently
            BufferedReader br = new BufferedReader(new FileReader(fileName));
            String line;

            // Read the file line by line
            while ((line = br.readLine()) != null) {
                lines++;                        // Count lines
                chars += line.length();          // Count characters
                String[] wordList = line.trim().split("\\s+"); // Split by spaces
                if (!line.trim().isEmpty()) {    // Avoid counting empty lines
                    words += wordList.length;    // Count words
                }
            }

            // Close the reader
            br.close();

            // Display results
            System.out.println("File: " + fileName);
            System.out.println("Number of lines: " + lines);
            System.out.println("Number of words: " + words);
            System.out.println("Number of characters: " + chars);
        } 
        catch (FileNotFoundException e) {
            System.out.println("File not found! Please check the file path.");
        } 
        catch (IOException e) {
            System.out.println("Error reading file: " + e.getMessage());
        }
    }
}
