// Enter the character from user using InputStreamReader 

     // File: ReadCharacter.java
import java.io.*;  // For InputStreamReader and IOException

public class ReadCharacter {
    public static void main(String[] args) throws IOException {
        // Create InputStreamReader to read bytes and decode into characters
        InputStreamReader reader = new InputStreamReader(System.in);

        System.out.print("Enter a character: ");
        
        // Read a single character from keyboard
        int inputChar = reader.read();

        // Display the character entered
        System.out.println("You entered: " + (char) inputChar);

        // Close the reader
        reader.close();
    }
}

OUTPUT

   
Enter a character: A
You entered: A
