//Enter the string from user using BufferedStreamReader 

 import java.io.*;

public class ReadStringExample {
    public static void main(String[] args) throws IOException {
        // Create InputStreamReader to read bytes and convert to characters
        InputStreamReader isr = new InputStreamReader(System.in);
        
        // Create BufferedReader for efficient reading of text
        BufferedReader br = new BufferedReader(isr);
        
        // Prompt user for input
        System.out.print("Enter a string: ");
        String input = br.readLine();  // Read string input
        
        // Display the entered string
        System.out.println("You entered: " + input);
        
        // Close the BufferedReader
        br.close();
    }
}
