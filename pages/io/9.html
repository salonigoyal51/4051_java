//Get all characters from the file and display on the screen 

 import java.io.*;

public class ReadFileCharacters {
    public static void main(String[] args) {
        try {
            // Create a FileReader object to read characters from a file
            FileReader fr = new FileReader("input.txt");

            int ch; // variable to hold character code

            System.out.println("Contents of the file:");

            // Read until end of file (-1)
            while ((ch = fr.read()) != -1) {
                System.out.print((char) ch); // convert int to char and display
            }

            // Close the file
            fr.close();
        } 
        catch (FileNotFoundException e) {
            System.out.println("File not found! Please check the file name or path.");
        } 
        catch (IOException e) {
            System.out.println("Error reading the file: " + e.getMessage());
        }
    }
}
