//Create a simple AWT application that displays a TextField with Custom Layout. 

   import java.awt.*;
import java.awt.event.*;

public class CustomLayoutTextFieldExample extends Frame {

    Label lbl;
    TextField tf;

    // Constructor
    CustomLayoutTextFieldExample() {
        // Disable layout manager for manual positioning
        setLayout(null);

        // Create Label
        lbl = new Label("Enter your name:");
        lbl.setBounds(50, 50, 120, 30);

        // Create TextField
        tf = new TextField();
        tf.setBounds(180, 50, 150, 30);

        // Add components to frame
        add(lbl);
        add(tf);

        // Frame settings
        setTitle("Custom Layout TextField Example");
        setSize(400, 200);
        setVisible(true);

        // Close window properly
        addWindowListener(new WindowAdapter() {
            public void windowClosing(WindowEvent e) {
                dispose();
            }
        });
    }

    // Main method
    public static void main(String[] args) {
        new CustomLayoutTextFieldExample();
    }
}
