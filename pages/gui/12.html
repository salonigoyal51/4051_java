//Create a simple AWT application that displays a CheckboxGroup with FlowLayout manager 

  import java.awt.*;
import java.awt.event.*;

public class FlowLayoutCheckboxGroupExample extends Frame implements ActionListener {

    Label lbl, resultLbl;
    CheckboxGroup genderGroup;
    Checkbox male, female, other;
    Button showBtn;

    // Constructor
    FlowLayoutCheckboxGroupExample() {
        // Set FlowLayout manager
        setLayout(new FlowLayout());

        // Label
        lbl = new Label("Select your gender:");

        // Create CheckboxGroup (radio button behavior)
        genderGroup = new CheckboxGroup();
        male = new Checkbox("Male", genderGroup, false);
        female = new Checkbox("Female", genderGroup, false);
        other = new Checkbox("Other", genderGroup, false);

        // Button
        showBtn = new Button("Show Selection");
        showBtn.addActionListener(this);

        // Label to display result
        resultLbl = new Label("Selected: None");

        // Add components to frame (FlowLayout automatically arranges them)
        add(lbl);
        add(male);
        add(female);
        add(other);
        add(showBtn);
        add(resultLbl);

        // Frame settings
        setTitle("FlowLayout CheckboxGroup Example");
        setSize(300, 200);
        setVisible(true);

        // Close window properly
        addWindowListener(new WindowAdapter() {
            public void windowClosing(WindowEvent e) {
                dispose();
            }
        });
    }

    // Handle button click
    public void actionPerformed(ActionEvent e) {
        Checkbox selected = genderGroup.getSelectedCheckbox();
        if (selected != null)
            resultLbl.setText("Selected: " + selected.getLabel());
        else
            resultLbl.setText("No selection made!");
    }

    // Main method
    public static void main(String[] args) {
        new FlowLayoutCheckboxGroupExample();
    }
}
