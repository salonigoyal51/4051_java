//Create a simple AWT application that displays a Choice with Custom Layout. 

  import java.awt.*;
import java.awt.event.*;

public class CustomLayoutChoiceExample extends Frame implements ActionListener {

    Label lbl, resultLbl;
    Choice colorChoice;
    Button showBtn;

    // Constructor
    CustomLayoutChoiceExample() {
        // Disable layout manager — using absolute positioning
        setLayout(null);

        // Label
        lbl = new Label("Select a color:");
        lbl.setBounds(50, 50, 120, 30);

        // Choice (Drop-down list)
        colorChoice = new Choice();
        colorChoice.setBounds(180, 50, 120, 30);
        colorChoice.add("Red");
        colorChoice.add("Green");
        colorChoice.add("Blue");
        colorChoice.add("Yellow");
        colorChoice.add("Black");

        // Button
        showBtn = new Button("Show Selection");
        showBtn.setBounds(100, 100, 150, 30);
        showBtn.addActionListener(this);

        // Label to show selected choice
        resultLbl = new Label("Selected: None");
        resultLbl.setBounds(50, 150, 200, 30);

        // Add components to frame
        add(lbl);
        add(colorChoice);
        add(showBtn);
        add(resultLbl);

        // Frame settings
        setTitle("Custom Layout Choice Example");
        setSize(350, 250);
        setVisible(true);

        // Close window properly
        addWindowListener(new WindowAdapter() {
            public void windowClosing(WindowEvent e) {
                dispose();
            }
        });
    }

    // Handle button click
    public void actionPerformed(ActionEvent e) {
        String selectedColor = colorChoice.getSelectedItem();
        resultLbl.setText("Selected: " + selectedColor);
    }

    // Main method
    public static void main(String[] args) {
        new CustomLayoutChoiceExample();
    }
}
