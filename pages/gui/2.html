//Create a simple AWT application that displays a button with Custom Layout 

    import java.awt.*;
import java.awt.event.*;

public class CustomLayoutButtonExample extends Frame implements ActionListener {

    Button btn;

    // Constructor
    CustomLayoutButtonExample() {
        // Disable default layout manager â€” use manual positioning
        setLayout(null);

        // Create Button
        btn = new Button("Click Me");
        btn.setBounds(100, 80, 100, 40); // x, y, width, height
        btn.addActionListener(this);

        // Add button to frame
        add(btn);

        // Frame settings
        setTitle("Custom Layout Button Example");
        setSize(300, 250);
        setVisible(true);

        // Close window properly
        addWindowListener(new WindowAdapter() {
            public void windowClosing(WindowEvent e) {
                dispose();
            }
        });
    }

    // Handle button click
    public void actionPerformed(ActionEvent e) {
        btn.setLabel("Clicked!");
    }

    // Main method
    public static void main(String[] args) {
        new CustomLayoutButtonExample();
    }
}
