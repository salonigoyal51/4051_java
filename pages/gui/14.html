//Create a simple AWT application that displays a List with FlowLayout manager. 

  import java.awt.*;
import java.awt.event.*;

public class FlowLayoutListExample extends Frame implements ActionListener {

    Label lbl;
    List fruitList;
    Button showBtn;
    TextField resultField;

    // Constructor
    FlowLayoutListExample() {
        // Set FlowLayout manager
        setLayout(new FlowLayout());

        // Create components
        lbl = new Label("Select your favorite fruits:");

        fruitList = new List(5, true); // 5 visible rows, multiple selection allowed
        fruitList.add("Apple");
        fruitList.add("Banana");
        fruitList.add("Orange");
        fruitList.add("Mango");
        fruitList.add("Grapes");

        showBtn = new Button("Show Selection");
        resultField = new TextField(25);
        resultField.setEditable(false);

        // Add components to the frame
        add(lbl);
        add(fruitList);
        add(showBtn);
        add(resultField);

        // Add action listener to button
        showBtn.addActionListener(this);

        // Frame settings
        setTitle("FlowLayout List Example");
        setSize(350, 300);
        setVisible(true);

        // Close window
        addWindowListener(new WindowAdapter() {
            public void windowClosing(WindowEvent e) {
                dispose();
            }
        });
    }

    // Handle button click
    public void actionPerformed(ActionEvent e) {
        String[] selectedItems = fruitList.getSelectedItems();
        String result = String.join(", ", selectedItems);
        resultField.setText("Selected: " + result);
    }

    // Main method
    public static void main(String[] args) {
        new FlowLayoutListExample();
    }
}
