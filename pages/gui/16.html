//Create a simple AWT application that displays a TextArea with FlowLayout manager 

  import java.awt.*;
import java.awt.event.*;

public class FlowLayoutTextAreaExample extends Frame {

    TextArea textArea;
    Label label;
    Button button;

    // Constructor
    FlowLayoutTextAreaExample() {
        // Set FlowLayout
        setLayout(new FlowLayout());

        // Create components
        label = new Label("Enter your text:");
        textArea = new TextArea(5, 30); // rows, columns
        button = new Button("Submit");

        // Add components to the frame
        add(label);
        add(textArea);
        add(button);

        // Frame properties
        setTitle("FlowLayout TextArea Example");
        setSize(400, 250);
        setVisible(true);

        // Close window on click
        addWindowListener(new WindowAdapter() {
            public void windowClosing(WindowEvent e) {
                dispose();
            }
        });
    }

    // Main method
    public static void main(String[] args) {
        new FlowLayoutTextAreaExample();
    }
}
