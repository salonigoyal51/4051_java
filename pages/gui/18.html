//Create a simple AWT application design a calculator that displays buttons, TextField with GridLayout manager. 

   import java.awt.*;
import java.awt.event.*;

public class SimpleCalculator extends Frame implements ActionListener {

    // Components
    TextField tf;
    Button[] numberButtons = new Button[10];
    Button add, sub, mul, div, eq, clr;
    Panel panel;

    double num1 = 0, num2 = 0, result = 0;
    char operator;

    // Constructor
    SimpleCalculator() {
        // Text field
        tf = new TextField();
        tf.setEditable(false);
        tf.setFont(new Font("Arial", Font.BOLD, 18));

        // Buttons
        add = new Button("+");
        sub = new Button("-");
        mul = new Button("*");
        div = new Button("/");
        eq = new Button("=");
        clr = new Button("C");

        // Number buttons 0-9
        for (int i = 0; i < 10; i++) {
            numberButtons[i] = new Button(String.valueOf(i));
            numberButtons[i].addActionListener(this);
        }

        // Operator buttons
        Button[] ops = {add, sub, mul, div, eq, clr};
        for (Button b : ops) {
            b.addActionListener(this);
        }

        // Panel with GridLayout (4 rows x 4 columns)
        panel = new Panel();
        panel.setLayout(new GridLayout(4, 4, 5, 5));

        // Add buttons to panel
        panel.add(numberButtons[7]);
        panel.add(numberButtons[8]);
        panel.add(numberButtons[9]);
        panel.add(add);
        panel.add(numberButtons[4]);
        panel.add(numberButtons[5]);
        panel.add(numberButtons[6]);
        panel.add(sub);
        panel.add(numberButtons[1]);
        panel.add(numberButtons[2]);
        panel.add(numberButtons[3]);
        panel.add(mul);
        panel.add(numberButtons[0]);
        panel.add(clr);
        panel.add(eq);
        panel.add(div);

        // Frame layout
        setLayout(new BorderLayout(5, 5));
        add(tf, BorderLayout.NORTH);
        add(panel, BorderLayout.CENTER);

        // Frame settings
        setTitle("Simple AWT Calculator");
        setSize(300, 400);
        setVisible(true);

        // Close window
        addWindowListener(new WindowAdapter() {
            public void windowClosing(WindowEvent e) {
                dispose();
            }
        });
    }

    // Action events
    public void actionPerformed(ActionEvent e) {
        String cmd = e.getActionCommand();

        if (cmd.charAt(0) >= '0' && cmd.charAt(0) <= '9') {
            tf.setText(tf.getText() + cmd);
        } else if (cmd.equals("+") || cmd.equals("-") || cmd.equals("*") || cmd.equals("/")) {
            num1 = Double.parseDouble(tf.getText());
            operator = cmd.charAt(0);
            tf.setText("");
        } else if (cmd.equals("=")) {
            num2 = Double.parseDouble(tf.getText());
            switch (operator) {
                case '+': result = num1 + num2; break;
                case '-': result = num1 - num2; break;
                case '*': result = num1 * num2; break;
                case '/': result = num1 / num2; break;
            }
            tf.setText(String.valueOf(result));
        } else if (cmd.equals("C")) {
            tf.setText("");
            num1 = num2 = result = 0;
        }
    }

    // Main method
    public static void main(String[] args) {
        new SimpleCalculator();
    }
}
