//Create a simple AWT application that displays a List with Custom Layout. 

 import java.awt.*;
import java.awt.event.*;

public class CustomLayoutListExample extends Frame implements ActionListener {

    Label lbl;
    List itemList;
    Button showBtn;
    TextField resultField;

    // Constructor
    CustomLayoutListExample() {
        // Disable default layout (use absolute positioning)
        setLayout(null);

        // Label
        lbl = new Label("Select your favorite fruits:");
        lbl.setBounds(50, 50, 200, 30);

        // List (multi-select)
        itemList = new List(5, true);  // 5 visible rows, multiple selection allowed
        itemList.setBounds(50, 90, 150, 100);
        itemList.add("Apple");
        itemList.add("Banana");
        itemList.add("Orange");
        itemList.add("Mango");
        itemList.add("Grapes");

        // Button
        showBtn = new Button("Show Selection");
        showBtn.setBounds(220, 100, 120, 30);
        showBtn.addActionListener(this);

        // TextField to display selected items
        resultField = new TextField();
        resultField.setBounds(50, 210, 290, 30);
        resultField.setEditable(false);

        // Add components to frame
        add(lbl);
        add(itemList);
        add(showBtn);
        add(resultField);

        // Frame settings
        setTitle("Custom Layout List Example");
        setSize(400, 300);
        setVisible(true);

        // Window close event
        addWindowListener(new WindowAdapter() {
            public void windowClosing(WindowEvent e) {
                dispose();
            }
        });
    }

    // Handle button click
    public void actionPerformed(ActionEvent e) {
        String[] selectedItems = itemList.getSelectedItems();
        String result = String.join(", ", selectedItems);
        resultField.setText("Selected: " + result);
    }

    // Main method
    public static void main(String[] args) {
        new CustomLayoutListExample();
    }
}
