//creates two threads to find and print even and odd numbers from 1 to 20. 

  // File: EvenOddThreads.java

class EvenThread extends Thread {
    public void run() {
        System.out.println("Even Thread Started:");
        for (int i = 2; i <= 20; i += 2) {
            System.out.println("Even: " + i);
            try {
                Thread.sleep(200); // small delay to simulate concurrent execution
            } catch (InterruptedException e) {
                e.printStackTrace();
            }
        }
        System.out.println("Even Thread Finished.\n");
    }
}

class OddThread extends Thread {
    public void run() {
        System.out.println("Odd Thread Started:");
        for (int i = 1; i <= 20; i += 2) {
            System.out.println("Odd: " + i);
            try {
                Thread.sleep(200); // small delay
            } catch (InterruptedException e) {
                e.printStackTrace();
            }
        }
        System.out.println("Odd Thread Finished.\n");
    }
}

public class EvenOddThreads {
    public static void main(String[] args) {
        // Create thread objects
        EvenThread even = new EvenThread();
        OddThread odd = new OddThread();

        // Start both threads
        even.start();
        odd.start();
    }
}
