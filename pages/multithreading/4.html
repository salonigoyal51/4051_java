//program that performs matrix multiplication using multiple threads. 

  // File: MatrixMultiplicationMultiThread.java

class MatrixMultiplicationThread extends Thread {
    private int[][] A, B, C;
    private int row; // row index to calculate

    public MatrixMultiplicationThread(int[][] A, int[][] B, int[][] C, int row) {
        this.A = A;
        this.B = B;
        this.C = C;
        this.row = row;
    }

    public void run() {
        int n = B[0].length; // number of columns in B
        int m = B.length;    // number of rows in B
        for (int j = 0; j < n; j++) {
            C[row][j] = 0;
            for (int k = 0; k < m; k++) {
                C[row][j] += A[row][k] * B[k][j];
            }
        }
        System.out.println("Thread for Row " + row + " completed.");
    }
}

public class MatrixMultiplicationMultiThread {
    public static void main(String[] args) throws InterruptedException {
        // Define matrix A (3x3)
        int[][] A = {
            {1, 2, 3},
            {4, 5, 6},
            {7, 8, 9}
        };

        // Define matrix B (3x3)
        int[][] B = {
            {9, 8, 7},
            {6, 5, 4},
            {3, 2, 1}
        };

        int rowsA = A.length;
        int colsB = B[0].length;
        int[][] C = new int[rowsA][colsB]; // Result matrix

        // Create one thread per row of matrix A
        MatrixMultiplicationThread[] threads = new MatrixMultiplicationThread[rowsA];
        for (int i = 0; i < rowsA; i++) {
            threads[i] = new MatrixMultiplicationThread(A, B, C, i);
            threads[i].start();
        }

        // Wait for all threads to finish
        for (int i = 0; i < rowsA; i++) {
            threads[i].join();
        }

        // Display result matrix
        System.out.println("\nResultant Matrix (A × B):");
        for (int i = 0; i < C.length; i++) {
            for (int j = 0; j < C[0].length; j++) {
                System.out.print(C[i][j] + "\t");
            }
            System.out.println();
        }
    }
}
