//calculate the sum of all prime numbers up to a given limit using multiple threads. 

 // File: PrimeSumMultiThread.java

class PrimeSumThread extends Thread {
    private int start, end;
    private long sum = 0;

    public PrimeSumThread(int start, int end) {
        this.start = start;
        this.end = end;
    }

    // Check if a number is prime
    private boolean isPrime(int n) {
        if (n <= 1) return false;
        for (int i = 2; i * i <= n; i++) {
            if (n % i == 0) return false;
        }
        return true;
    }

    // Thread execution
    public void run() {
        for (int i = start; i <= end; i++) {
            if (isPrime(i)) {
                sum += i;
            }
        }
        System.out.println("Thread " + Thread.currentThread().getName() +
                " processed range " + start + "–" + end + " | Partial Sum = " + sum);
    }

    public long getSum() {
        return sum;
    }
}

public class PrimeSumMultiThread {
    public static void main(String[] args) throws InterruptedException {
        int limit = 1000;          // You can change this or take user input
        int numThreads = 4;        // Number of threads
        int range = limit / numThreads;

        PrimeSumThread[] threads = new PrimeSumThread[numThreads];

        int start = 1;
        int end;

        // Create and start threads
        for (int i = 0; i < numThreads; i++) {
            end = (i == numThreads - 1) ? limit : start + range - 1;
            threads[i] = new PrimeSumThread(start, end);
            threads[i].start();
            start = end + 1;
        }

        // Wait for all threads to complete
        long totalSum = 0;
        for (int i = 0; i < numThreads; i++) {
            threads[i].join();
            totalSum += threads[i].getSum();
        }

        // Display result
        System.out.println("\n✅ Total Sum of Prime Numbers up to " + limit + " = " + totalSum);
    }
}
