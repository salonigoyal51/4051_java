//create a Animal interface with a method called bark() that takes no arguments and returns void. 

  // File: AnimalInterfaceExample.java

// Step 1: Define the interface
interface Animal {
    void bark();  // Method with no arguments and no return value
}

// Step 2: Create Dog class that implements Animal
class Dog implements Animal {
    @Override
    public void bark() {
        System.out.println("Dog is barking");
    }
}

// Step 3: Main class to test the program
public class AnimalInterfaceExample {
    public static void main(String[] args) {
        // Create an object of Dog
        Animal myDog = new Dog();

        // Call the bark() method
        myDog.bark();
    }
}

OUTPUT

  Dog is barking
