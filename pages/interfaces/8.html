PROGRAM

// Write a Java program to create an interface Playable with a method play() that takes no arguments and returns void. 

  // File: PlayableExample.java

// Step 1: Define the interface
interface Playable {
    void play();  // Abstract method
}

// Step 2: Implement Playable in Football class
class Football implements Playable {
    @Override
    public void play() {
        System.out.println("Playing Football! Kick the ball and score goals!");
    }
}

// Step 3: Implement Playable in Volleyball class
class Volleyball implements Playable {
    @Override
    public void play() {
        System.out.println("Playing Volleyball! Serve, spike, and block the ball!");
    }
}

// Step 4: Implement Playable in Basketball class
class Basketball implements Playable {
    @Override
    public void play() {
        System.out.println("Playing Basketball! Dribble, shoot, and dunk!");
    }
}

// Step 5: Main class to test all implementations
public class PlayableExample {
    public static void main(String[] args) {
        // Create interface references for each sport
        Playable football = new Football();
        Playable volleyball = new Volleyball();
        Playable basketball = new Basketball();

        // Call play() method for each
        football.play();
        volleyball.play();
        basketball.play();

        // OR use polymorphism through an array
        System.out.println("\nUsing Polymorphism with Array:");
        Playable[] games = { new Football(), new Volleyball(), new Basketball() };
        for (Playable game : games) {
            game.play();
        }
    }
}
