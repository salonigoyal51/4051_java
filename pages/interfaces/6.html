//create an interface Drawable with a method draw() that takes no arguments and returns void. 

   
    // File: DrawableExample.java

// Step 1: Define the interface
interface Drawable {
    void draw(); // Abstract method
}

// Step 2: Implement Drawable in Circle class
class Circle implements Drawable {
    @Override
    public void draw() {
        System.out.println("Drawing a Circle.");
    }
}

// Step 3: Implement Drawable in Rectangle class
class Rectangle implements Drawable {
    @Override
    public void draw() {
        System.out.println("Drawing a Rectangle.");
    }
}

// Step 4: Implement Drawable in Triangle class
class Triangle implements Drawable {
    @Override
    public void draw() {
        System.out.println("Drawing a Triangle.");
    }
}

// Step 5: Main class to test
public class DrawableExample {
    public static void main(String[] args) {
        // Create objects of each shape
        Drawable circle = new Circle();
        Drawable rectangle = new Rectangle();
        Drawable triangle = new Triangle();

        // Call the draw() method for each shape
        circle.draw();
        rectangle.draw();
        triangle.draw();

        // OR demonstrate polymorphism using an array
        System.out.println("\nUsing polymorphism:");
        Drawable[] shapes = { new Circle(), new Rectangle(), new Triangle() };
        for (Drawable shape : shapes) {
            shape.draw();
        }
    }
}
