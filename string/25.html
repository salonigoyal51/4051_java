//Program to find the duplicate words in a string 

      
     import java.util.HashMap;
import java.util.Scanner;

public class DuplicateWords {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // Input string
        System.out.print("Enter a string: ");
        String str = sc.nextLine().toLowerCase(); // convert to lowercase for case-insensitive comparison

        // Split the string into words
        String[] words = str.split("\\s+"); // split by spaces

        // Count frequency of each word
        HashMap wordCount = new HashMap<>();
        for (String word : words) {
            if (wordCount.containsKey(word)) {
                wordCount.put(word, wordCount.get(word) + 1);
            } else {
                wordCount.put(word, 1);
            }
        }

        // Display duplicate words
        System.out.println("Duplicate words in the string are:");
        boolean found = false;
        for (String word : wordCount.keySet()) {
            if (wordCount.get(word) > 1) {
                System.out.println(word + " appears " + wordCount.get(word) + " times");
                found = true;
            }
        }

        if (!found) {
            System.out.println("No duplicate words found.");
        }

        sc.close();
    }
}
