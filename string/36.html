// print smallest and biggest possible palindrome word in a given string 

 import java.util.Scanner;

public class PalindromeWords {
    // Method to check if a word is a palindrome
    public static boolean isPalindrome(String word) {
        int i = 0, j = word.length() - 1;
        while (i < j) {
            if (word.charAt(i) != word.charAt(j)) {
                return false;
            }
            i++;
            j--;
        }
        return true;
    }

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // Input string
        System.out.print("Enter a string: ");
        String str = sc.nextLine();

        String[] words = str.split("\\s+"); // split into words

        String smallest = null;
        String biggest = null;

        for (String word : words) {
            String cleanWord = word.replaceAll("[^a-zA-Z0-9]", ""); // remove punctuation
            if (cleanWord.length() > 0 && isPalindrome(cleanWord)) {
                if (smallest == null || cleanWord.length() < smallest.length()) {
                    smallest = cleanWord;
                }
                if (biggest == null || cleanWord.length() > biggest.length()) {
                    biggest = cleanWord;
                }
            }
        }

        if (smallest != null && biggest != null) {
            System.out.println("Smallest palindrome word: " + smallest);
            System.out.println("Biggest palindrome word: " + biggest);
        } else {
            System.out.println("No palindrome words found in the string.");
        }

        sc.close();
    }
}
