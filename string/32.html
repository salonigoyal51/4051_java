//Program to Insert a string into another string 

  import java.util.Scanner;

public class InsertString {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // Input the original string
        System.out.print("Enter the original string: ");
        String original = sc.nextLine();

        // Input the string to insert
        System.out.print("Enter the string to insert: ");
        String toInsert = sc.nextLine();

        // Input the position where to insert
        System.out.print("Enter the position to insert at (0-based index): ");
        int position = sc.nextInt();

        // Check if position is valid
        if (position < 0 || position > original.length()) {
            System.out.println("Invalid position! Position should be between 0 and " + original.length());
        } else {
            // Use StringBuilder to insert
            StringBuilder sb = new StringBuilder(original);
            sb.insert(position, toInsert);

            System.out.println("Resulting string: " + sb.toString());
        }

        sc.close();
    }
}
