//Program to Convert Enum to String 

    // Define an enum
enum Day {
    MONDAY, TUESDAY, WEDNESDAY, THURSDAY, FRIDAY, SATURDAY, SUNDAY
}

public class EnumToString {
    public static void main(String[] args) {
        // Get an enum value
        Day today = Day.FRIDAY;

        // Convert enum to String
        String str1 = today.name();       // returns the name of enum constant
        String str2 = today.toString();   // also returns the name (can be overridden in enum)

        // Display results
        System.out.println("Using name(): " + str1);
        System.out.println("Using toString(): " + str2);
    }
}

OUTPUT

   Using name(): FRIDAY
Using toString(): FRIDAY
