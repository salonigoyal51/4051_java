// Program to divide a string in 'N' equal parts 

   import java.util.Scanner;

public class DivideString {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // Input string
        System.out.print("Enter a string: ");
        String str = sc.nextLine();

        // Input number of parts
        System.out.print("Enter the number of parts (N): ");
        int N = sc.nextInt();

        int length = str.length();

        // Check if string can be divided equally
        if (length % N != 0) {
            System.out.println("The string cannot be divided into " + N + " equal parts.");
        } else {
            int partSize = length / N;

            System.out.println("Dividing string into " + N + " equal parts:");

            for (int i = 0; i < length; i += partSize) {
                String part = str.substring(i, i + partSize);
                System.out.println(part);
            }
        }

        sc.close();
    }
}
