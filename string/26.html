//Program to find the frequency of characters 

  import java.util.Scanner;

public class CharacterFrequency {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // Input string
        System.out.print("Enter a string: ");
        String str = sc.nextLine().toLowerCase(); // case-insensitive

        int[] freq = new int[256]; // frequency array for ASCII characters

        // Count frequency of each character
        for (int i = 0; i < str.length(); i++) {
            char ch = str.charAt(i);
            freq[ch]++;
        }

        // Display frequency of each character
        System.out.println("Frequency of characters:");
        for (int i = 0; i < 256; i++) {
            if (freq[i] > 0) {
                System.out.println((char) i + " : " + freq[i]);
            }
        }

        sc.close();
    }
}
