//Program to Convert String to String Array 

   import java.util.Arrays;
import java.util.Scanner;

public class StringToArray {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // Input string from user
        System.out.print("Enter a string: ");
        String input = sc.nextLine();

        // Convert string to string array using space as delimiter
        String[] arr = input.split(" "); // splits the string at spaces

        // Display the array
        System.out.println("String Array: " + Arrays.toString(arr));

        sc.close();
    }
}

OUTPUT

Enter a string: Java is fun
