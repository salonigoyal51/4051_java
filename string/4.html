//Program to remove all occurrences of a given character from input String 

   
     import java.util.Scanner;

public class RemoveCharacter {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // Input string from user
        System.out.print("Enter a string: ");
        String input = sc.nextLine();

        // Input character to remove
        System.out.print("Enter the character to remove: ");
        char ch = sc.next().charAt(0);

        // Remove all occurrences of the character
        String result = input.replace(String.valueOf(ch), "");

        // Display result
        System.out.println("String after removing '" + ch + "': " + result);

        sc.close();
    }
}
