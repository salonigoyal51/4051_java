// replace lower-case characters with upper-case and vice-versa 

  import java.util.Scanner;

public class SwapCase {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // Input string
        System.out.print("Enter a string: ");
        String str = sc.nextLine();

        StringBuilder swapped = new StringBuilder();

        // Swap case for each character
        for (int i = 0; i < str.length(); i++) {
            char ch = str.charAt(i);
            if (Character.isLowerCase(ch)) {
                swapped.append(Character.toUpperCase(ch));
            } else if (Character.isUpperCase(ch)) {
                swapped.append(Character.toLowerCase(ch));
            } else {
                swapped.append(ch); // keep non-alphabetic characters unchanged
            }
        }

        System.out.println("String after swapping case: " + swapped.toString());

        sc.close();
    }
}
