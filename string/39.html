//Program to replace the spaces of a string with a specific character 

 import java.util.Scanner;

public class ReplaceSpaces {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // Input string
        System.out.print("Enter a string: ");
        String str = sc.nextLine();

        // Input the character to replace spaces with
        System.out.print("Enter a character to replace spaces with: ");
        char replacement = sc.next().charAt(0);

        // Replace spaces with the given character
        String result = str.replace(' ', replacement);

        System.out.println("String after replacing spaces: " + result);

        sc.close();
    }
}

OUTPUT
