//count the total number of punctuation characters exists in a String 

   import java.util.Scanner;

public class PunctuationCount {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // Input string from user
        System.out.print("Enter a string: ");
        String input = sc.nextLine();

        int count = 0;

        // Check each character
        for (int i = 0; i < input.length(); i++) {
            char ch = input.charAt(i);
            // If character is not a letter, digit, or space, consider it punctuation
            if (!Character.isLetterOrDigit(ch) && !Character.isWhitespace(ch)) {
                count++;
            }
        }

        System.out.println("Total number of punctuation characters: " + count);

        sc.close();
    }
}
