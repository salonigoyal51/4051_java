//Program to Add Characters to a String 

    import java.util.Scanner;

public class AddCharacterToString {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // Input original string
        System.out.print("Enter original string: ");
        String str = sc.nextLine();

        // Input character(s) to add
        System.out.print("Enter character(s) to add: ");
        String charsToAdd = sc.nextLine();

        // Add characters to the string
        str = str + charsToAdd;  // or use str.concat(charsToAdd);

        // Display the result
        System.out.println("String after adding characters: " + str);

        sc.close();
    }
}
