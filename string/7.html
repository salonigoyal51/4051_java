//Program to Add Characters to a String 

   
    import java.util.Scanner;

public class AddCharactersToString {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // Input original string
        System.out.print("Enter the original string: ");
        String str = sc.nextLine();

        // Input character(s) to add
        System.out.print("Enter character(s) to add: ");
        String toAdd = sc.nextLine();

        // Add character(s) to the string
        str = str + toAdd; // or str.concat(toAdd);

        // Display result
        System.out.println("String after adding: " + str);

        sc.close();
    }
}
