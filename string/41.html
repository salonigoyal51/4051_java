//Program to Splitting into a number of sub-strings 

 import java.util.Scanner;

public class SplitString {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // Input string
        System.out.print("Enter a string: ");
        String str = sc.nextLine();

        // --- 1. Splitting using a delimiter (e.g., space) ---
        System.out.println("\nSplitting string by spaces:");
        String[] words = str.split("\\s+"); // split by one or more spaces
        for (String word : words) {
            System.out.println(word);
        }

        // --- 2. Splitting into fixed-size substrings ---
        System.out.print("\nEnter the length of each substring: ");
        int partLength = sc.nextInt();

        System.out.println("Splitting string into substrings of length " + partLength + ":");
        for (int i = 0; i < str.length(); i += partLength) {
            // Handle last substring if it's shorter than partLength
            int end = Math.min(i + partLength, str.length());
            System.out.println(str.substring(i, end));
        }

        sc.close();
    }
}
