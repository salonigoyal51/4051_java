//Program to Iterate Over Characters in String 

  import java.util.Scanner;

public class IterateStringCharacters {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // Input string from user
        System.out.print("Enter a string: ");
        String str = sc.nextLine();

        System.out.println("Iterating using standard for loop:");
        for (int i = 0; i < str.length(); i++) {
            System.out.println("Character at index " + i + " : " + str.charAt(i));
        }

        System.out.println("\nIterating using for-each loop with toCharArray():");
        for (char ch : str.toCharArray()) {
            System.out.println(ch);
        }

        System.out.println("\nIterating using Java 8 forEach and lambda:");
        str.chars().forEach(c -> System.out.println((char) c));

        sc.close();
    }
}
