//Program to find the most repeated word in a text file 

  import java.io.File;
import java.io.FileNotFoundException;
import java.util.HashMap;
import java.util.Scanner;
import java.util.Map;

public class MostRepeatedWord {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        System.out.print("Enter the path of the text file: ");
        String filePath = sc.nextLine();

        HashMap wordCount = new HashMap<>();

        try {
            Scanner fileScanner = new Scanner(new File(filePath));

            while (fileScanner.hasNextLine()) {
                String line = fileScanner.nextLine().toLowerCase(); // case-insensitive
                // Remove punctuation
                line = line.replaceAll("[^a-zA-Z0-9\\s]", "");

                // Split line into words
                String[] words = line.split("\\s+");

                for (String word : words) {
                    if (!word.isEmpty()) {
                        wordCount.put(word, wordCount.getOrDefault(word, 0) + 1);
                    }
                }
            }

            fileScanner.close();

            // Find the most repeated word
            String mostRepeated = null;
            int maxCount = 0;

            for (Map.Entry entry : wordCount.entrySet()) {
                if (entry.getValue() > maxCount) {
                    maxCount = entry.getValue();
                    mostRepeated = entry.getKey();
                }
            }

            if (mostRepeated != null) {
                System.out.println("Most repeated word: " + mostRepeated);
                System.out.println("Frequency: " + maxCount);
            } else {
                System.out.println("No words found in the file.");
            }

        } catch (FileNotFoundException e) {
            System.out.println("File not found: " + filePath);
        }

        sc.close();
    }
}
