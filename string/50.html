// Create an object StringBuffer class to show functionality of insert, append, delete and replace functions. 

  import java.util.Scanner;

public class StringBufferFunctions {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // Input initial string
        System.out.print("Enter initial string: ");
        String str = sc.nextLine();

        // Create StringBuffer object
        StringBuffer sb = new StringBuffer(str);
        System.out.println("Original StringBuffer: " + sb);

        // 1. Append a string
        System.out.print("Enter string to append: ");
        String appendStr = sc.nextLine();
        sb.append(appendStr);
        System.out.println("After append: " + sb);

        // 2. Insert a string at a specific index
        System.out.print("Enter string to insert: ");
        String insertStr = sc.nextLine();
        System.out.print("Enter index to insert at: ");
        int insertIndex = sc.nextInt();
        sc.nextLine(); // consume newline
        if (insertIndex >= 0 && insertIndex <= sb.length()) {
            sb.insert(insertIndex, insertStr);
            System.out.println("After insert: " + sb);
        } else {
            System.out.println("Invalid index for insert!");
        }

        // 3. Delete a substring
        System.out.print("Enter start index to delete: ");
        int deleteStart = sc.nextInt();
        System.out.print("Enter end index to delete: ");
        int deleteEnd = sc.nextInt();
        sc.nextLine(); // consume newline
        if (deleteStart >= 0 && deleteEnd <= sb.length() && deleteStart < deleteEnd) {
            sb.delete(deleteStart, deleteEnd);
            System.out.println("After delete: " + sb);
        } else {
            System.out.println("Invalid indices for delete!");
        }

        // 4. Replace a substring
        System.out.print("Enter start index to replace: ");
        int replaceStart = sc.nextInt();
        System.out.print("Enter end index to replace: ");
        int replaceEnd = sc.nextInt();
        sc.nextLine(); // consume newline
        System.out.print("Enter replacement string: ");
        String replaceStr = sc.nextLine();
        if (replaceStart >= 0 && replaceEnd <= sb.length() && replaceStart < replaceEnd) {
            sb.replace(replaceStart, replaceEnd, replaceStr);
            System.out.println("After replace: " + sb);
        } else {
            System.out.println("Invalid indices for replace!");
        }

        sc.close();
    }
}
