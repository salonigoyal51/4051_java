//Program to find the duplicate characters in a string 

   import java.util.Scanner;

public class DuplicateCharacters {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // Input string from user
        System.out.print("Enter a string: ");
        String str = sc.nextLine().toLowerCase(); // case-insensitive

        int[] freq = new int[256]; // Array to store frequency of each character

        // Count frequency of each character
        for (int i = 0; i < str.length(); i++) {
            char ch = str.charAt(i);
            freq[ch]++;
        }

        System.out.println("Duplicate characters in the string are:");
        boolean found = false;
        for (int i = 0; i < 256; i++) {
            if (freq[i] > 1) {
                System.out.println((char) i + " appears " + freq[i] + " times");
                found = true;
            }
        }

        if (!found) {
            System.out.println("No duplicate characters found.");
        }

        sc.close();
    }
}
