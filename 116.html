// print out all Armstrong numbers between 1 to 600 using loop 

   public class ArmstrongNumbers {
    public static void main(String[] args) {

        System.out.println("Armstrong numbers between 1 and 600 are:");

        // Loop through numbers 1 to 600
        for (int num = 1; num <= 600; num++) {
            int originalNum = num;
            int sum = 0;
            int digits = 0;
            int temp = num;

            // Count number of digits
            while (temp > 0) {
                digits++;
                temp = temp / 10;
            }

            temp = num;

            // Calculate sum of cubes/power of digits
            while (temp > 0) {
                int remainder = temp % 10;
                sum += Math.pow(remainder, digits);
                temp = temp / 10;
            }

            // Check if Armstrong
            if (sum == originalNum) {
                System.out.println(originalNum);
            }
        }
    }
}
