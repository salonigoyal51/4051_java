//find the frequency of odd & even numbers in the given matrix 

    import java.util.Scanner;

public class OddEvenFrequencyMatrix {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // Take number of rows and columns
        System.out.print("Enter number of rows: ");
        int rows = sc.nextInt();

        System.out.print("Enter number of columns: ");
        int cols = sc.nextInt();

        int[][] matrix = new int[rows][cols];
        int evenCount = 0, oddCount = 0;

        // Input elements of the matrix
        System.out.println("Enter elements of the matrix:");
        for (int i = 0; i < rows; i++) {
            for (int j = 0; j < cols; j++) {
                matrix[i][j] = sc.nextInt();

                // Count even and odd numbers
                if (matrix[i][j] % 2 == 0) {
                    evenCount++;
                } else {
                    oddCount++;
                }
            }
        }

        // Display matrix
        System.out.println("\nThe Matrix is:");
        for (int i = 0; i < rows; i++) {
            for (int j = 0; j < cols; j++) {
                System.out.print(matrix[i][j] + " ");
            }
            System.out.println();
        }

        // Display frequency
        System.out.println("\nFrequency of even numbers: " + evenCount);
        System.out.println("Frequency of odd numbers: " + oddCount);

        sc.close();
    }
}
