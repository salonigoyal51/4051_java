//How to convert Object to String 

   public class ObjectToStringExample {
    public static void main(String[] args) {
        Object obj = "Hello Java";        // Object value (actually holds a String)

        // Convert Object to String
        String str = obj.toString();      // using toString() method

        System.out.println("Object value: " + obj);
        System.out.println("String value: " + str);
    }
}

OUTPUT

   Object value: Hello Java
String value: Hello Java
