//How to convert int to char 

   public class IntToCharExample {
    public static void main(String[] args) {
        int num = 65;  // ASCII value for 'A'

        // Convert int to char
        char ch = (char) num;

        System.out.println("Integer value: " + num);
        System.out.println("Character: " + ch);
    }
}

OUTPUT

   Integer value: 65
Character: A
