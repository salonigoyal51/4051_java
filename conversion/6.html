//How to convert float to String 

    public class FloatToStringExample {
    public static void main(String[] args) {
        float number = 12.34f;                // float value
        String str = String.valueOf(number);  // convert float to String

        System.out.println("Float value: " + number);
        System.out.println("String value: " + str);
    }
}

OUTPUT

   Float value: 12.34
String value: 12.34
