//convert String to int 

   import java.util.Scanner;

public class StringToInt {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // Take a numeric string input from user
        System.out.print("Enter a number (in string form): ");
        String str = sc.nextLine();

        // Convert String to int using Integer.parseInt()
        int num = Integer.parseInt(str);

        // Display the result
        System.out.println("String value: " + str);
        System.out.println("Integer value: " + num);

        // Perform some operation to show itâ€™s an integer
        System.out.println("After adding 10: " + (num + 10));

        sc.close();
    }
}
