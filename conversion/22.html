//How to convert boolean to String 

    public class BooleanToStringExample {
    public static void main(String[] args) {
        boolean b1 = true;
        boolean b2 = false;

        // Convert boolean to String
        String str1 = String.valueOf(b1);
        String str2 = Boolean.toString(b2);

        System.out.println("Boolean: " + b1 + " → String: " + str1);
        System.out.println("Boolean: " + b2 + " → String: " + str2);
    }
}

OUTPUT

   Boolean: true → String: true
Boolean: false → String: false
