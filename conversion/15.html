//How to convert int to long 

   public class IntToLongExample {
    public static void main(String[] args) {
        int num = 100;     // int value

        // Convert int to long
        long result = num;  // implicit type conversion (widening)

        System.out.println("Integer value: " + num);
        System.out.println("Long value: " + result);
    }
}

OUTPUT

   Integer value: 100
Long value: 100
