//How to convert String to char 

    public class StringToCharExample {
    public static void main(String[] args) {
        String str = "Hello";  // String value

        // Convert String to char (get single character)
        char ch = str.charAt(0); // get first character

        System.out.println("String value: " + str);
        System.out.println("First character: " + ch);
    }
}

OUTPUT

   
   String value: Hello
First character: H
