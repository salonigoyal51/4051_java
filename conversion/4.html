//convert long to String 

   
    public class LongToStringExample {
    public static void main(String[] args) {
        long number = 9876543210L;        // long value
        String str = String.valueOf(number); // convert long to String

        System.out.println("Long value: " + number);
        System.out.println("String value: " + str);
    }
}

OUTPUT

   Long value: 9876543210
String value: 9876543210
