//How to convert double to int 

   public class DoubleToIntExample {
    public static void main(String[] args) {
        double num = 25.78;   // double value

        // Convert double to int (explicit casting)
        int result = (int) num;

        System.out.println("Double value: " + num);
        System.out.println("Integer value: " + result);
    }
}

OUTPUT

   Double value: 25.78
Integer value: 25
