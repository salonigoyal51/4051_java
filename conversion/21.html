//How to convert String to boolean 

    public class StringToBooleanExample {
    public static void main(String[] args) {
        String str1 = "true";
        String str2 = "false";
        String str3 = "TrUe"; // case insensitive

        // Convert String to boolean
        boolean b1 = Boolean.parseBoolean(str1);
        boolean b2 = Boolean.parseBoolean(str2);
        boolean b3 = Boolean.parseBoolean(str3);

        System.out.println("String: " + str1 + " → boolean: " + b1);
        System.out.println("String: " + str2 + " → boolean: " + b2);
        System.out.println("String: " + str3 + " → boolean: " + b3);
    }
}

OUTPUT

   String: true → boolean: true
String: false → boolean: false
String: TrUe → boolean: true
