//How to convert int to double 

   public class IntToDoubleExample {
    public static void main(String[] args) {
        int num = 25;          // int value

        // Convert int to double (automatic conversion)
        double result = num;

        System.out.println("Integer value: " + num);
        System.out.println("Double value: " + result);
    }
}

OUTPUT

  Integer value: 25
Double value: 25.0
