//How to convert double to String 

    
    public class DoubleToStringExample {
    public static void main(String[] args) {
        double number = 45.67;                // double value
        String str = String.valueOf(number);  // convert double to String

        System.out.println("Double value: " + number);
        System.out.println("String value: " + str);
    }
}

OUTPUT

    
  Double value: 45.67
String value: 45.67
