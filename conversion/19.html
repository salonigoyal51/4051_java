//How to convert char to int 

   public class CharToIntExample {
    public static void main(String[] args) {
        char ch = 'A';  // character value

        // Convert char to int
        int asciiValue = ch;

        System.out.println("Character: " + ch);
        System.out.println("ASCII value: " + asciiValue);
    }
}

OUTPUT

   Character: A
ASCII value: 65
