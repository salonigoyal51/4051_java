//create a package with multiple default classes. 

 Folder Structure
 mypackage/
    Addition.java
    Subtraction.java
    Calculator.java

    Addition.java
    // File: Addition.java
// Step 1: Declare the package
package mypackage;

// Step 2: Define the Addition class
class Addition {
    public int add(int a, int b) {
        return a + b;
    }
}
// File: Subtraction.java
package mypackage;

class Subtraction {
    public int subtract(int a, int b) {
        return a - b;
    }
}
// File: Calculator.java
package mypackage;

public class Calculator {
    public static void main(String[] args) {
        // Create objects of Addition and Subtraction classes
        Addition addObj = new Addition();
        Subtraction subObj = new Subtraction();

        int a = 10, b = 5;

        System.out.println("Addition of " + a + " and " + b + " = " + addObj.add(a, b));
        System.out.println("Subtraction of " + a + " and " + b + " = " + subObj.subtract(a, b));
    }
}

OUTPUT

 Addition of 10 and 5 = 15
Subtraction of 10 and 5 = 5
Compilation and Execution Steps

In terminal or command prompt:
cd path_to_your_source_folder
javac mypackage/*.java
java mypackage.Calculator
