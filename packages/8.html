//create a package with public class and protected members to be accessed in another class. 

  // File: Person.java
package mypackage;

public class Person {
    // Protected members â€” accessible in subclasses even outside the package
    protected String name;
    protected int age;

    // Public method to set values
    public void setPerson(String n, int a) {
        name = n;
        age = a;
    }

    // Protected method (can be accessed by subclass)
    protected void displayPerson() {
        System.out.println("Name: " + name);
        System.out.println("Age: " + age);
    }
}
// File: AccessProtected.java
import mypackage.Person;

// Create subclass to access protected members
class Employee extends Person {
    private String designation;

    public void setEmployeeDetails(String n, int a, String d) {
        setPerson(n, a);  // Call public method from parent
        designation = d;
    }

    public void showEmployeeDetails() {
        System.out.println("===== Employee Details =====");
        // Access protected members (inherited from Person)
        System.out.println("Name: " + name);
        System.out.println("Age: " + age);
        System.out.println("Designation: " + designation);
    }
}

// Main class to run the program
public class AccessProtected {
    public static void main(String[] args) {
        Employee emp = new Employee();
        emp.setEmployeeDetails("Anuradha", 28, "Software Engineer");
        emp.showEmployeeDetails();
    }
}
