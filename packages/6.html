// File: Addition.java
package mypackage;

public class Addition {
    public int add(int a, int b) {
        return a + b;
    }
}
// File: Subtraction.java
package mypackage;

public class Subtraction {
    public int subtract(int a, int b) {
        return a - b;
    }
}
// File: Multiplication.java
package mypackage;

public class Multiplication {
    public int multiply(int a, int b) {
        return a * b;
    }
}
// File: Division.java
package mypackage;

public class Division {
    public double divide(int a, int b) {
        if (b == 0) {
            System.out.println("Division by zero is not allowed!");
            return 0;
        }
        return (double) a / b;
    }
}
// File: MainApp.java
import mypackage.*;  // Import all public classes from mypackage

public class MainApp {
    public static void main(String[] args) {
        // Create objects of each class
        Addition addObj = new Addition();
        Subtraction subObj = new Subtraction();
        Multiplication mulObj = new Multiplication();
        Division divObj = new Division();

        int a = 20, b = 5;

        // Call methods
        System.out.println("Addition of " + a + " and " + b + " = " + addObj.add(a, b));
        System.out.println("Subtraction of " + a + " and " + b + " = " + subObj.subtract(a, b));
        System.out.println("Multiplication of " + a + " and " + b + " = " + mulObj.multiply(a, b));
        System.out.println("Division of " + a + " and " + b + " = " + divObj.divide(a, b));
    }
}

OUTPUT

   Addition of 20 and 5 = 25
Subtraction of 20 and 5 = 15
Multiplication of 20 and 5 = 100
Division of 20 and 5 = 4.0
How to Compile and Run
Step 1️⃣ – Compile the package classes
javac mypackage/*.java
Step 2️⃣ – Compile the main program
javac -cp . MainApp.java
Step 3️⃣ – Run the main class
java -cp . MainApp
