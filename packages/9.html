//Create a nested package i.e. c4.sample with public class and protected members to be accessed in another class. 

 // File: Data.java
package c4.sample;

public class Data {
    // Protected members
    protected String message = "Hello from protected member!";
    protected int number = 42;

    // Protected method
    protected void displayData() {
        System.out.println("Message: " + message);
        System.out.println("Number: " + number);
    }
}
// File: AccessNested.java
import c4.sample.Data;  // Import the nested package class

// Subclass of Data to access protected members
class DerivedData extends Data {
    public void showInfo() {
        System.out.println("=== Accessing Protected Members from Subclass ===");
        // Accessing inherited protected fields and method
        System.out.println("Message: " + message);
        System.out.println("Number: " + number);
        displayData(); // calling protected method
    }
}

// Main class
public class AccessNested {
    public static void main(String[] args) {
        DerivedData obj = new DerivedData();
        obj.showInfo();
    }
}
