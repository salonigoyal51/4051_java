//program to access the java.util.Date class through qualified way and print the current date of the system but inherit the Date class. 

// File: QualifiedInheritedDateExample.java

// No import statement used

// Step 1: Create a subclass that inherits from java.util.Date
class MyDate extends java.util.Date {
    // Method to display the current date and time
    public void displayDate() {
        System.out.println("Current system date and time: " + this);
    }
}

// Step 2: Main class to test inheritance
public class QualifiedInheritedDateExample {
    public static void main(String[] args) {
        // Create an object of the subclass
        MyDate current = new MyDate();

        // Display the current date and time
        current.displayDate();
    }
}

OUTPUT

  Current system date and time: Mon Oct 20 19:55:45 IST 2025
