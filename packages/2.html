//program to access the java.util.Date class through qualified way and print the current date of the system 

 // File: QualifiedDateExample.java

public class QualifiedDateExample {
    public static void main(String[] args) {
        // Using fully qualified class name (no import statement)
        java.util.Date currentDate = new java.util.Date();

        // Print the current system date and time
        System.out.println("Current system date and time: " + currentDate);
    }
}

OUTPUT

 Current system date and time: Mon Oct 20 19:45:10 IST 2025
