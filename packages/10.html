//Create a package with public class and public members to be accessed in another class. 

  // File: Student.java
package mypackage;  // Define package name

public class Student {
    // Public members
    public String name;
    public int rollNo;
    public double marks;

    // Public method to display student details
    public void display() {
        System.out.println("=== Student Details ===");
        System.out.println("Name: " + name);
        System.out.println("Roll No: " + rollNo);
        System.out.println("Marks: " + marks);
    }
}
// File: AccessPublic.java
import mypackage.Student;  // Importing the package class

public class AccessPublic {
    public static void main(String[] args) {
        // Create an object of Student class
        Student s1 = new Student();

        // Access public members directly
        s1.name = "Anuradha";
        s1.rollNo = 101;
        s1.marks = 95.5;

        // Call public method
        s1.display();
    }
}

OUTPUT

  === Student Details ===
Name: Anuradha
Roll No: 101
Marks: 95.5
