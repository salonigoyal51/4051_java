//program to access the java.util.Date class through import keyword and print the current date of the system. 

// File: DateExample.java

// Step 1: Import the Date class from java.util package
import java.util.Date;

// Step 2: Define the main class
public class DateExample {
    public static void main(String[] args) {
        // Step 3: Create a Date object to get current date and time
        Date currentDate = new Date();

        // Step 4: Print the current date
        System.out.println("Current system date and time: " + currentDate);
    }
}

OUTPUT

   Current system date and time: Mon Oct 20 19:35:22 IST 2025
