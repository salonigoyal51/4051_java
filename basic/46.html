//Program To Calculate CGPA Percentage 

   import java.util.Scanner;

public class CalculateCGPA {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // Input number of subjects
        System.out.print("Enter the number of subjects: ");
        int n = sc.nextInt();

        double totalGradePoints = 0;

        // Input grade points for each subject
        for (int i = 1; i <= n; i++) {
            System.out.print("Enter grade point for subject " + i + ": ");
            double grade = sc.nextDouble();
            totalGradePoints += grade;
        }

        // Calculate CGPA
        double cgpa = totalGradePoints / n;

        // Convert CGPA to percentage (Approximation: Percentage = CGPA Ã— 9.5)
        double percentage = cgpa * 9.5;

        // Display results
        System.out.println("\nYour CGPA = " + cgpa);
        System.out.println("Equivalent Percentage = " + percentage + "%");

        sc.close();
    }
}
