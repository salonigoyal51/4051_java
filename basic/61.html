//display the n terms of odd natural number and their sum 

import java.util.Scanner;

public class OddNumbersSum {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // Input number of terms
        System.out.print("Enter the number of terms: ");
        int n = sc.nextInt();

        int sum = 0;

        System.out.println("\nThe first " + n + " odd natural numbers are:");

        // Loop to print odd numbers and calculate sum
        for (int i = 1; i <= n; i++) {
            int odd = 2 * i - 1;
            System.out.println(odd);
            sum += odd;
        }

        System.out.println("\nSum of first " + n + " odd natural numbers = " + sum);

        sc.close();
    }
}
