PROGRAM

//program to check whether Java is installed on your computer or not 

    public class CheckJavaInstallation {
    public static void main(String[] args) {
        // Print Java version details
        System.out.println("Checking Java installation...");

        String version = System.getProperty("java.version");
        String vendor = System.getProperty("java.vendor");
        String home = System.getProperty("java.home");

        if (version != null) {
            System.out.println("\n✅ Java is installed on your system!");
            System.out.println("Java Version: " + version);
            System.out.println("Java Vendor: " + vendor);
            System.out.println("Java Installation Path: " + home);
        } else {
            System.out.println("\n❌ Java is not installed on your system.");
        }
    }
}
