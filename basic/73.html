//program to get a substring of a given string between two specified positions 

   import java.util.Scanner;

public class SubstringExample {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        System.out.print("Enter a string: ");
        String str = sc.nextLine();

        // Input start and end positions
        System.out.print("Enter starting position: ");
        int start = sc.nextInt();

        System.out.print("Enter ending position: ");
        int end = sc.nextInt();

        // Validate input
        if (start >= 0 && end <= str.length() && start < end) {
            // Get substring
            String subStr = str.substring(start, end);
            System.out.println("Substring from position " + start + " to " + end + " is: " + subStr);
        } else {
            System.out.println("Invalid positions! Please enter valid start and end values.");
        }

        sc.close();
    }
}
