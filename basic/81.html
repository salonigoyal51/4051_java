PROGRAM

//program to count total number of lines from a string 

  public class CountLinesInString {
    public static void main(String[] args) {
        // Sample multiline string
        String text = "This is line one.\nThis is line two.\nThis is line three.";

        // Split the string by newline character
        String[] lines = text.split("\n");

        // Count number of lines
        int numberOfLines = lines.length;

        // Display result
        System.out.println("The text is:\n" + text);
        System.out.println("\nTotal number of lines: " + numberOfLines);
    }
}
