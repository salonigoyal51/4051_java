//program to remove html tags from a string 

   import java.util.Scanner;

public class RemoveHTMLTags {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // Input string containing HTML tags
        System.out.print("Enter a string with HTML tags: ");
        String html = sc.nextLine();

        // Remove HTML tags using regular expression
        String plainText = html.replaceAll("<[^>]*>", "");

        // Display result
        System.out.println("Original String: " + html);
        System.out.println("String after removing HTML tags: " + plainText);

        sc.close();
    }
}
