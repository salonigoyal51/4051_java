//program to find surface area of cube 

   import java.util.Scanner;

public class SurfaceAreaOfCube {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // Input side of the cube
        System.out.print("Enter the side length of the cube: ");
        double side = sc.nextDouble();

        // Calculate surface area using formula: 6 × side²
        double surfaceArea = 6 * Math.pow(side, 2);

        // Display result
        System.out.println("Surface area of the cube = " + surfaceArea);

        sc.close();
    }
}
