//display the cube of the number upto given an integer 

   import java.util.Scanner;

public class CubeOfNumbers {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // Input number
        System.out.print("Enter the number of terms: ");
        int n = sc.nextInt();

        System.out.println("\nNumber\tCube");
        System.out.println("-----------------");

        // Loop to display cube of each number
        for (int i = 1; i <= n; i++) {
            int cube = i * i * i;
            System.out.println(i + "\t" + cube);
        }

        sc.close();
    }
}
