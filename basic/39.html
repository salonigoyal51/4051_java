//program to find surface area of cylinder 

  import java.util.Scanner;

public class SurfaceAreaOfCylinder {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // Input radius and height
        System.out.print("Enter the radius of the cylinder: ");
        double radius = sc.nextDouble();

        System.out.print("Enter the height of the cylinder: ");
        double height = sc.nextDouble();

        // Calculate surface area using formula: 2Ï€r(h + r)
        double surfaceArea = 2 * Math.PI * radius * (height + radius);

        // Display result
        System.out.println("Surface area of the cylinder = " + surfaceArea);

        sc.close();
    }
}
