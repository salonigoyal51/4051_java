//Program To Calculate Batting Average 

   import java.util.Scanner;

public class BattingAverage {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // Input total runs scored
        System.out.print("Enter total runs scored: ");
        int runs = sc.nextInt();

        // Input number of times player got out
        System.out.print("Enter number of times player got out: ");
        int outs = sc.nextInt();

        // Check for division by zero
        if (outs == 0) {
            System.out.println("Batting average cannot be calculated (player never got out).");
        } else {
            double average = (double) runs / outs;
            System.out.println("Batting Average = " + average);
        }

        sc.close();
    }
}
