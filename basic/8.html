//program to convert a binary number to decimal number 

  import java.util.Scanner;

public class BinaryToDecimal {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // Input binary number as a string
        System.out.print("Enter a binary number: ");
        String binary = sc.nextLine();

        // Convert binary to decimal using built-in method
        int decimal = Integer.parseInt(binary, 2);

        // Display result
        System.out.println("Decimal equivalent of binary " + binary + " is: " + decimal);

        sc.close();
    }
}

OUTPUT

   Enter a binary number: 1010
Decimal equivalent of binary 1010 is: 10
