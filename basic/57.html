//program to input 5 numbers from keyboard and find their sum and average 

   import java.util.Scanner;

public class SumAndAverage {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        int sum = 0;
        double average;
        int number;

        // Input 5 numbers and calculate sum
        System.out.println("Enter 5 numbers:");
        for (int i = 1; i <= 5; i++) {
            System.out.print("Number " + i + ": ");
            number = sc.nextInt();
            sum += number;
        }

        // Calculate average
        average = sum / 5.0;

        // Display results
        System.out.println("\nSum of 5 numbers = " + sum);
        System.out.println("Average of 5 numbers = " + average);

        sc.close();
    }
}
