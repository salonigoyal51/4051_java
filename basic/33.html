//Program to find area of Prism 

   import java.util.Scanner;

public class AreaOfPrism {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // Input dimensions
        System.out.print("Enter the length of the prism: ");
        double length = sc.nextDouble();

        System.out.print("Enter the width of the prism: ");
        double width = sc.nextDouble();

        System.out.print("Enter the height of the prism: ");
        double height = sc.nextDouble();

        // Calculate surface area
        double area = 2 * (length * width + width * height + height * length);

        // Display result
        System.out.println("Surface area of the prism = " + area);

        sc.close();
    }
}
