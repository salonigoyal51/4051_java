//Program to calculate electricity bill 

   import java.util.Scanner;

public class ElectricityBill {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // Input the number of units consumed
        System.out.print("Enter total units consumed: ");
        double units = sc.nextDouble();

        double bill = 0;

        // Calculate bill according to unit slabs
        if (units <= 100) {
            bill = units * 1.20; // ₹1.20 per unit for first 100 units
        } 
        else if (units <= 300) {
            bill = 100 * 1.20 + (units - 100) * 2.00; // ₹2 per unit for next 200 units
        } 
        else {
            bill = 100 * 1.20 + 200 * 2.00 + (units - 300) * 3.00; // ₹3 per unit above 300
        }

        // Add fixed meter charge
        double totalBill = bill + 50; // ₹50 fixed charge

        // Display result
        System.out.println("Electricity Bill = ₹" + totalBill);

        sc.close();
    }
}
