//find second lowest number from the array 

  import java.util.Scanner;
import java.util.Arrays;

public class SecondLowest {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // Input array size
        System.out.print("Enter the number of elements: ");
        int n = sc.nextInt();

        if (n < 2) {
            System.out.println("Array should contain at least two elements.");
            return;
        }

        int[] arr = new int[n];

        // Input array elements
        System.out.println("Enter " + n + " integers:");
        for (int i = 0; i < n; i++) {
            arr[i] = sc.nextInt();
        }

        // Sort the array
        Arrays.sort(arr);

        // Find the second lowest (smallest unique) number
        int secondLowest = -1;
        for (int i = 1; i < n; i++) {
            if (arr[i] != arr[0]) {
                secondLowest = arr[i];
                break;
            }
        }

        if (secondLowest == -1)
            System.out.println("All elements are the same.");
        else
            System.out.println("The second lowest number is: " + secondLowest);

        sc.close();
    }
}
