//program that accepts an integer (n) and computes the value of n+nn+nnn 

   import java.util.Scanner;

public class ComputeN_NN_NNN {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // Input integer n
        System.out.print("Enter an integer (n): ");
        int n = sc.nextInt();

        // Create nn and nnn using string concatenation
        String nnStr = Integer.toString(n) + Integer.toString(n);
        String nnnStr = nnStr + Integer.toString(n);

        // Convert back to integers
        int nn = Integer.parseInt(nnStr);
        int nnn = Integer.parseInt(nnnStr);

        // Compute result
        int result = n + nn + nnn;

        // Display results
        System.out.println("\nn = " + n);
        System.out.println("nn = " + nn);
        System.out.println("nnn = " + nnn);
        System.out.println("Result (n + nn + nnn) = " + result);

        sc.close();
    }
}
