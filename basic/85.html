//program to test if an array contains a specific value 

   import java.util.Scanner;

public class CheckArrayValue {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // Input array size
        System.out.print("Enter number of elements in the array: ");
        int n = sc.nextInt();

        int[] arr = new int[n];

        // Input array elements
        System.out.println("Enter " + n + " elements:");
        for (int i = 0; i < n; i++) {
            arr[i] = sc.nextInt();
        }

        // Input value to check
        System.out.print("Enter the value to search: ");
        int key = sc.nextInt();

        boolean found = false;

        // Check if array contains the value
        for (int num : arr) {
            if (num == key) {
                found = true;
                break;
            }
        }

        // Display result
        if (found)
            System.out.println("The array contains the value " + key + ".");
        else
            System.out.println("The array does not contain the value " + key + ".");

        sc.close();
    }
}
