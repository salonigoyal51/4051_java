//display the first 5 natural numbers 

   public class FirstFiveNaturalNumbers {
    public static void main(String[] args) {

        System.out.println("The first 5 natural numbers are:");

        for (int i = 1; i <= 5; i++) {
            System.out.println(i);
        }
    }
}
