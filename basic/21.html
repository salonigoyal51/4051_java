//program to swap the first and last elements of an array and create a new array 

   import java.util.Scanner;

public class SwapFirstLast {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // Input array size
        System.out.print("Enter the number of elements in the array: ");
        int n = sc.nextInt();

        int[] arr = new int[n];
        int[] newArr = new int[n];  // new array for swapped result

        // Input array elements
        System.out.println("Enter the elements of the array:");
        for (int i = 0; i < n; i++) {
            arr[i] = sc.nextInt();
        }

        // Copy all elements to new array
        for (int i = 0; i < n; i++) {
            newArr[i] = arr[i];
        }

        // Swap first and last element in the new array
        if (n > 1) {
            int temp = newArr[0];
            newArr[0] = newArr[n - 1];
            newArr[n - 1] = temp;
        }

        // Display new array
        System.out.println("\nArray after swapping first and last elements:");
        for (int i = 0; i < n; i++) {
            System.out.print(newArr[i] + " ");
        }

        sc.close();
    }
}
